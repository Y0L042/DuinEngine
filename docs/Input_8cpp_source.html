114void ProcessSDLKeyboardEvent(::SDL_Event e)
115{
116 if (!EVENT_IS_KEYBOARD(e.type))
117 {
118 return;
119 }
120 // Set current key state
121 ::SDL_Scancode code = e.key.scancode;
122 if (code > 511 || code < 0)
123 {
124 DN_CORE_WARN("Keyboard event out of bounds {}!", (int)code);
125 return;
126 }
127
128 KeyState state = KeyState::UP;
129 if (e.type == SDL_EVENT_KEY_DOWN)
130 {
131 state = KeyState::DOWN;
132 }
133 if (e.type == SDL_EVENT_KEY_UP)
134 {
135 state = KeyState::UP;
136 }
137 currentKeyState[code] = state;
138}

184// Expand a modifier mask: replace each combined alias (CTRL, SHIFT, ALT, GUI) with
185// any of its individual components that are present in `active`.
186// This allows DN_KEY_MOD_CTRL to match when either LCTRL or RCTRL is active,
187// while keeping LCTRL and RCTRL bindings side-specific.
188static DN_Keymod ExpandBinding(DN_Keymod binding, DN_Keymod active)
189{
190 DN_Keymod expanded = binding;
191 // For each combined pair: if the binding requests the combined bit but not a specific
192 // side, replace the combined bits with whichever individual bits are active.
193 auto expandPair = [&](DN_Keymod combined, DN_Keymod left, DN_Keymod right) {
194 DN_Keymod both = left | right;
195 if ((binding & both) == both) {
196 // Binding uses the combined alias (both bits set) — satisfied by any active side
197 if (active & left) expanded = (expanded & ~both) | left;
198 else if (active & right) expanded = (expanded & ~both) | right;
199 }
200 };
201 expandPair(DN_KEY_MOD_CTRL, DN_KEY_MOD_LCTRL, DN_KEY_MOD_RCTRL);
202 expandPair(DN_KEY_MOD_SHIFT, DN_KEY_MOD_LSHIFT, DN_KEY_MOD_RSHIFT);
203 expandPair(DN_KEY_MOD_ALT, DN_KEY_MOD_LALT, DN_KEY_MOD_RALT);
204 expandPair(DN_KEY_MOD_GUI, DN_KEY_MOD_LGUI, DN_KEY_MOD_RGUI);
205 return expanded;
206}

220int IsModifierDown(DN_Scancode code)
221{
222 DN_Keymod mod = DN_KEY_MOD_NONE;
223 switch (code)
224 {
225 case DN_SCANCODE_LCTRL: mod = DN_KEY_MOD_LCTRL; break;
226 case DN_SCANCODE_RCTRL: mod = DN_KEY_MOD_RCTRL; break;
227 case DN_SCANCODE_LSHIFT: mod = DN_KEY_MOD_LSHIFT; break;
228 case DN_SCANCODE_RSHIFT: mod = DN_KEY_MOD_RSHIFT; break;
229 case DN_SCANCODE_LALT: mod = DN_KEY_MOD_LALT; break;
230 case DN_SCANCODE_RALT: mod = DN_KEY_MOD_RALT; break;
231 case DN_SCANCODE_LGUI: mod = DN_KEY_MOD_LGUI; break;
232 case DN_SCANCODE_RGUI: mod = DN_KEY_MOD_RGUI; break;
233 default: return 0;
234 }
235 return (currentModifierState & mod) != 0;
236}

243void ProcessSDLMouseEvent(::SDL_Event e)
244{
245 if (e.type == SDL_EVENT_MOUSE_MOTION)
246 {
247 float x = e.motion.x;
248 float y = e.motion.y;
249 currentMouseLocalPos = Vector2(x, y);
250
251 mouseDeltaX += e.motion.xrel;
252 mouseDeltaY += e.motion.yrel;
253 }
254
255 KeyState state = KeyState::UP;
256 if (e.type == SDL_EVENT_MOUSE_BUTTON_DOWN)
257 {
258 state = KeyState::DOWN;
259 DN_MouseButtonFlags btnIdx = e.button.button - 1;
260 currentMouseKeyState[btnIdx] = state;
261 }
262 if (e.type == SDL_EVENT_MOUSE_BUTTON_UP)
263 {
264 state = KeyState::UP;
265 DN_MouseButtonFlags btnIdx = e.button.button - 1;
266 currentMouseKeyState[btnIdx] = state;
267 }
268
269 if (e.type == SDL_EVENT_MOUSE_WHEEL)
270 {
271 DN_CORE_INFO("MOUSE_WHEEL {} {}", e.wheel.x, e.wheel.y);
272 mouseScrollDelta.x += e.wheel.x;
273 mouseScrollDelta.y += e.wheel.y;
274 }
