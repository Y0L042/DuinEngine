71void ProcessSDLKeyboardEvent(::SDL_Event e)
72{
73 if (!EVENT_IS_KEYBOARD(e.type))
74 {
75 return;
76 }
77 // Set current key state
78 ::SDL_Scancode code = e.key.scancode;
79 if (code > 511 || code < 0)
80 {
81 DN_CORE_WARN("Keyboard event out of bounds {}!", (int)code);
82 return;
83 }
84
85 KeyState state = KeyState::UP;
86 if (e.type == SDL_EVENT_KEY_DOWN)
87 {
88 state = KeyState::DOWN;
89 }
90 if (e.type == SDL_EVENT_KEY_UP)
91 {
92 state = KeyState::UP;
93 }
94 currentKeyState[code] = state;
95}

154void ProcessSDLMouseEvent(::SDL_Event e)
155{
156 previousMouseLocalPos = currentMouseLocalPos;
157
158 if (e.type == SDL_EVENT_MOUSE_MOTION)
159 {
160 float x, y = 0;
161 ::SDL_GetMouseState(&x, &y);
162 currentMouseLocalPos = Vector2(x, y);
163 }
164
165 KeyState state = KeyState::UP;
166 if (e.type == SDL_EVENT_MOUSE_BUTTON_DOWN)
167 {
168 state = KeyState::DOWN;
169 DN_MouseButtonFlags btnIdx = e.button.button - 1;
170 currentMouseKeyState[btnIdx] = state;
171 }
172 if (e.type == SDL_EVENT_MOUSE_BUTTON_UP)
173 {
174 state = KeyState::UP;
175 DN_MouseButtonFlags btnIdx = e.button.button - 1;
176 currentMouseKeyState[btnIdx] = state;
177 }
178
179 mouseScrollDelta = Vector2();
180 if (e.type == SDL_EVENT_MOUSE_WHEEL)
181 {
182 DN_CORE_INFO("MOUSE_WHEEL {} {}", e.wheel.x, e.wheel.y);
183 mouseScrollDelta.x = e.wheel.x;
184 mouseScrollDelta.y = e.wheel.y;
185 }
186}
