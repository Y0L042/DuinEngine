101 void Frustum(float left, float right, float bottom, float top, float znear, float zfar, float* m16)
102 {
103 float temp, temp2, temp3, temp4;
104 temp = 2.0f * znear;
105 temp2 = right - left;
106 temp3 = top - bottom;
107 temp4 = zfar - znear;
108 m16[0] = temp / temp2;
109 m16[1] = 0.0;
110 m16[2] = 0.0;
111 m16[3] = 0.0;
112 m16[4] = 0.0;
113 m16[5] = temp / temp3;
114 m16[6] = 0.0;
115 m16[7] = 0.0;
116 m16[8] = (right + left) / temp2;
117 m16[9] = (top + bottom) / temp3;
118 m16[10] = (-zfar - znear) / temp4;
119 m16[11] = -1.0f;
120 m16[12] = 0.0;
121 m16[13] = 0.0;
122 m16[14] = (-temp * zfar) / temp4;
123 m16[15] = 0.0;
124 }

154 void LookAt(const float* eye, const float* at, const float* up, float* m16)
155 {
156 float X[3], Y[3], Z[3], tmp[3];
157
158 tmp[0] = eye[0] - at[0];
159 tmp[1] = eye[1] - at[1];
160 tmp[2] = eye[2] - at[2];
161 Normalize(tmp, Z);
162 Normalize(up, Y);
163 Cross(Y, Z, tmp);
164 Normalize(tmp, X);
165 Cross(Z, X, tmp);
166 Normalize(tmp, Y);
167
168 m16[0] = X[0];
169 m16[1] = Y[0];
170 m16[2] = Z[0];
171 m16[3] = 0.0f;
172 m16[4] = X[1];
173 m16[5] = Y[1];
174 m16[6] = Z[1];
175 m16[7] = 0.0f;
176 m16[8] = X[2];
177 m16[9] = Y[2];
178 m16[10] = Z[2];
179 m16[11] = 0.0f;
180 m16[12] = -Dot(X, eye);
181 m16[13] = -Dot(Y, eye);
182 m16[14] = -Dot(Z, eye);
183 m16[15] = 1.0f;
184 }

458 float matrix_t::Inverse(const matrix_t& srcMatrix, bool affine)
459 {
460 float det = 0;
461
462 if (affine)
463 {
464 det = GetDeterminant();
465 float s = 1 / det;
466 m[0][0] = (srcMatrix.m[1][1] * srcMatrix.m[2][2] - srcMatrix.m[1][2] * srcMatrix.m[2][1]) * s;
467 m[0][1] = (srcMatrix.m[2][1] * srcMatrix.m[0][2] - srcMatrix.m[2][2] * srcMatrix.m[0][1]) * s;
468 m[0][2] = (srcMatrix.m[0][1] * srcMatrix.m[1][2] - srcMatrix.m[0][2] * srcMatrix.m[1][1]) * s;
469 m[1][0] = (srcMatrix.m[1][2] * srcMatrix.m[2][0] - srcMatrix.m[1][0] * srcMatrix.m[2][2]) * s;
470 m[1][1] = (srcMatrix.m[2][2] * srcMatrix.m[0][0] - srcMatrix.m[2][0] * srcMatrix.m[0][2]) * s;
471 m[1][2] = (srcMatrix.m[0][2] * srcMatrix.m[1][0] - srcMatrix.m[0][0] * srcMatrix.m[1][2]) * s;
472 m[2][0] = (srcMatrix.m[1][0] * srcMatrix.m[2][1] - srcMatrix.m[1][1] * srcMatrix.m[2][0]) * s;
473 m[2][1] = (srcMatrix.m[2][0] * srcMatrix.m[0][1] - srcMatrix.m[2][1] * srcMatrix.m[0][0]) * s;
474 m[2][2] = (srcMatrix.m[0][0] * srcMatrix.m[1][1] - srcMatrix.m[0][1] * srcMatrix.m[1][0]) * s;
475 m[3][0] = -(m[0][0] * srcMatrix.m[3][0] + m[1][0] * srcMatrix.m[3][1] + m[2][0] * srcMatrix.m[3][2]);
476 m[3][1] = -(m[0][1] * srcMatrix.m[3][0] + m[1][1] * srcMatrix.m[3][1] + m[2][1] * srcMatrix.m[3][2]);
477 m[3][2] = -(m[0][2] * srcMatrix.m[3][0] + m[1][2] * srcMatrix.m[3][1] + m[2][2] * srcMatrix.m[3][2]);
478 }
479 else
480 {
481 // transpose matrix
482 float src[16];
483 for (int i = 0; i < 4; ++i)
484 {
485 src[i] = srcMatrix.m16[i * 4];
486 src[i + 4] = srcMatrix.m16[i * 4 + 1];
487 src[i + 8] = srcMatrix.m16[i * 4 + 2];
488 src[i + 12] = srcMatrix.m16[i * 4 + 3];
489 }
490
491 // calculate pairs for first 8 elements (cofactors)
492 float tmp[12]; // temp array for pairs
493 tmp[0] = src[10] * src[15];
494 tmp[1] = src[11] * src[14];
495 tmp[2] = src[9] * src[15];
496 tmp[3] = src[11] * src[13];
497 tmp[4] = src[9] * src[14];
498 tmp[5] = src[10] * src[13];
499 tmp[6] = src[8] * src[15];
500 tmp[7] = src[11] * src[12];
501 tmp[8] = src[8] * src[14];
502 tmp[9] = src[10] * src[12];
503 tmp[10] = src[8] * src[13];
504 tmp[11] = src[9] * src[12];
505
506 // calculate first 8 elements (cofactors)
507 m16[0] = (tmp[0] * src[5] + tmp[3] * src[6] + tmp[4] * src[7]) - (tmp[1] * src[5] + tmp[2] * src[6] + tmp[5] * src[7]);
508 m16[1] = (tmp[1] * src[4] + tmp[6] * src[6] + tmp[9] * src[7]) - (tmp[0] * src[4] + tmp[7] * src[6] + tmp[8] * src[7]);
509 m16[2] = (tmp[2] * src[4] + tmp[7] * src[5] + tmp[10] * src[7]) - (tmp[3] * src[4] + tmp[6] * src[5] + tmp[11] * src[7]);
510 m16[3] = (tmp[5] * src[4] + tmp[8] * src[5] + tmp[11] * src[6]) - (tmp[4] * src[4] + tmp[9] * src[5] + tmp[10] * src[6]);
511 m16[4] = (tmp[1] * src[1] + tmp[2] * src[2] + tmp[5] * src[3]) - (tmp[0] * src[1] + tmp[3] * src[2] + tmp[4] * src[3]);
512 m16[5] = (tmp[0] * src[0] + tmp[7] * src[2] + tmp[8] * src[3]) - (tmp[1] * src[0] + tmp[6] * src[2] + tmp[9] * src[3]);
513 m16[6] = (tmp[3] * src[0] + tmp[6] * src[1] + tmp[11] * src[3]) - (tmp[2] * src[0] + tmp[7] * src[1] + tmp[10] * src[3]);
514 m16[7] = (tmp[4] * src[0] + tmp[9] * src[1] + tmp[10] * src[2]) - (tmp[5] * src[0] + tmp[8] * src[1] + tmp[11] * src[2]);
515
516 // calculate pairs for second 8 elements (cofactors)
517 tmp[0] = src[2] * src[7];
518 tmp[1] = src[3] * src[6];
519 tmp[2] = src[1] * src[7];
520 tmp[3] = src[3] * src[5];
521 tmp[4] = src[1] * src[6];
522 tmp[5] = src[2] * src[5];
523 tmp[6] = src[0] * src[7];
524 tmp[7] = src[3] * src[4];
525 tmp[8] = src[0] * src[6];
526 tmp[9] = src[2] * src[4];
527 tmp[10] = src[0] * src[5];
528 tmp[11] = src[1] * src[4];
529
530 // calculate second 8 elements (cofactors)
531 m16[8] = (tmp[0] * src[13] + tmp[3] * src[14] + tmp[4] * src[15]) - (tmp[1] * src[13] + tmp[2] * src[14] + tmp[5] * src[15]);
532 m16[9] = (tmp[1] * src[12] + tmp[6] * src[14] + tmp[9] * src[15]) - (tmp[0] * src[12] + tmp[7] * src[14] + tmp[8] * src[15]);
533 m16[10] = (tmp[2] * src[12] + tmp[7] * src[13] + tmp[10] * src[15]) - (tmp[3] * src[12] + tmp[6] * src[13] + tmp[11] * src[15]);
534 m16[11] = (tmp[5] * src[12] + tmp[8] * src[13] + tmp[11] * src[14]) - (tmp[4] * src[12] + tmp[9] * src[13] + tmp[10] * src[14]);
535 m16[12] = (tmp[2] * src[10] + tmp[5] * src[11] + tmp[1] * src[9]) - (tmp[4] * src[11] + tmp[0] * src[9] + tmp[3] * src[10]);
536 m16[13] = (tmp[8] * src[11] + tmp[0] * src[8] + tmp[7] * src[10]) - (tmp[6] * src[10] + tmp[9] * src[11] + tmp[1] * src[8]);
537 m16[14] = (tmp[6] * src[9] + tmp[11] * src[11] + tmp[3] * src[8]) - (tmp[10] * src[11] + tmp[2] * src[8] + tmp[7] * src[9]);
538 m16[15] = (tmp[10] * src[10] + tmp[4] * src[8] + tmp[9] * src[9]) - (tmp[8] * src[9] + tmp[11] * src[10] + tmp[5] * src[8]);
539
540 // calculate determinant
541 det = src[0] * m16[0] + src[1] * m16[1] + src[2] * m16[2] + src[3] * m16[3];
542
543 // calculate matrix inverse
544 float invdet = 1 / det;
545 for (int j = 0; j < 16; ++j)
546 {
547 m16[j] *= invdet;
548 }
549 }
550
551 return det;
552 }

554 void matrix_t::RotationAxis(const vec_t& axis, float angle)
555 {
556 float length2 = axis.LengthSq();
557 if (length2 < FLT_EPSILON)
558 {
559 SetToIdentity();
560 return;
561 }
562
563 vec_t n = axis * (1.f / sqrtf(length2));
564 float s = sinf(angle);
565 float c = cosf(angle);
566 float k = 1.f - c;
567
568 float xx = n.x * n.x * k + c;
569 float yy = n.y * n.y * k + c;
570 float zz = n.z * n.z * k + c;
571 float xy = n.x * n.y * k;
572 float yz = n.y * n.z * k;
573 float zx = n.z * n.x * k;
574 float xs = n.x * s;
575 float ys = n.y * s;
576 float zs = n.z * s;
577
578 m[0][0] = xx;
579 m[0][1] = xy + zs;
580 m[0][2] = zx - ys;
581 m[0][3] = 0.f;
582 m[1][0] = xy - zs;
583 m[1][1] = yy;
584 m[1][2] = yz + xs;
585 m[1][3] = 0.f;
586 m[2][0] = zx + ys;
587 m[2][1] = yz - xs;
588 m[2][2] = zz;
589 m[2][3] = 0.f;
590 m[3][0] = 0.f;
591 m[3][1] = 0.f;
592 m[3][2] = 0.f;
593 m[3][3] = 1.f;
594 }

848 static float GetSegmentLengthClipSpace(const vec_t& start, const vec_t& end, const bool localCoordinates = false)
849 {

874 static float GetParallelogram(const vec_t& ptO, const vec_t& ptA, const vec_t& ptB)
875 {
876 vec_t pts[] = { ptO, ptA, ptB };
877 for (unsigned int i = 0; i < 3; i++)
878 {
879 pts[i].TransformPoint(gContext.mMVP);
880 if (fabsf(pts[i].w) > FLT_EPSILON) // check for axis aligned with camera direction
881 {
882 pts[i] *= 1.f / pts[i].w;
883 }
884 }
885 vec_t segA = pts[1] - pts[0];
886 vec_t segB = pts[2] - pts[0];
887 segA.y /= gContext.mDisplayRatio;
888 segB.y /= gContext.mDisplayRatio;
889 vec_t segAOrtho = makeVect(-segA.y, segA.x);
890 segAOrtho.Normalize();
891 float dt = segAOrtho.Dot3(segB);
892 float surface = sqrtf(segA.x * segA.x + segA.y * segA.y) * fabsf(dt);
893 return surface;
894 }

896 inline vec_t PointOnSegment(const vec_t& point, const vec_t& vertPos1, const vec_t& vertPos2)
897 {
898 vec_t c = point - vertPos1;
899 vec_t V;
900
901 V.Normalize(vertPos2 - vertPos1);
902 float d = (vertPos2 - vertPos1).Length();
903 float t = V.Dot3(c);
904
905 if (t < 0.f)
906 {
907 return vertPos1;
908 }
909
910 if (t > d)
911 {
912 return vertPos2;
913 }
914
915 return vertPos1 + V * t;
916 }

1170 static void ComputeTripodAxisAndVisibility(const int axisIndex, vec_t& dirAxis, vec_t& dirPlaneX, vec_t& dirPlaneY, bool& belowAxisLimit, bool& belowPlaneLimit, const bool localCoordinates = false)
1171 {
1172 dirAxis = directionUnary[axisIndex];

1174 dirPlaneY = directionUnary[(axisIndex + 2) % 3];
1175
1176 if (gContext.mbUsing && (gContext.GetCurrentID() == gContext.mEditingID))

1226 static void ComputeSnap(float* value, float snap)
1227 {
1228 if (snap <= FLT_EPSILON)
1229 {
1230 return;
1231 }
1232
1233 float modulo = fmodf(*value, snap);
1234 float moduloRatio = fabsf(modulo) / snap;
1235 if (moduloRatio < snapTension)
1236 {
1237 *value -= modulo;
1238 }
1239 else if (moduloRatio > (1.f - snapTension))
1240 {
1241 *value = *value - modulo + snap * ((*value < 0.f) ? -1.f : 1.f);
1242 }
1243 }
1244 static void ComputeSnap(vec_t& value, const float* snap)
1245 {
1246 for (int i = 0; i < 3; i++)
1247 {
1248 ComputeSnap(&value[i], snap[i]);
1249 }
