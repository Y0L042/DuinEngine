53 JSONValue JSONValue::Parse(const std::string& string)
54 {
55 JSONValue dv;
56
57 dv.jdoc_ = std::make_shared<rapidjson::Document>();
58 dv.jdoc_->Parse(string.c_str());
59 dv.jvalue_ = dv.jdoc_.get();
60
61 // Check for parse errors
62 if (dv.jdoc_->HasParseError()) {
63 DN_CORE_ERROR("JSON parsing failed at offset {}: Error code {}",
64 dv.jdoc_->GetErrorOffset(),
65 static_cast<int>(dv.jdoc_->GetParseError()));
66 dv.SetObject(); // Default to empty object on error
67 return dv;
68 }
69
70 // Accept any valid JSON type (object, array, primitive, etc.)
71 if (dv.jvalue_->IsObject()) {
72 DN_CORE_TRACE("Parsed JSON object");
73 } else if (dv.jvalue_->IsArray()) {
74 DN_CORE_TRACE("Parsed JSON array");
75 } else if (dv.jvalue_->IsString()) {
76 DN_CORE_TRACE("Parsed JSON string");
77 } else if (dv.jvalue_->IsNumber()) {
78 DN_CORE_TRACE("Parsed JSON number");
79 } else if (dv.jvalue_->IsBool()) {
80 DN_CORE_TRACE("Parsed JSON boolean");
81 } else if (dv.jvalue_->IsNull()) {
82 DN_CORE_TRACE("Parsed JSON null");
83 }
84
85 return dv;
86 }

686 JSONValue::ConstDataIterator JSONValue::FindMember(const std::string& member) const
687 {
688 if (jvalue_) {
689 if (jvalue_->IsObject()) {
690 auto mit = jvalue_->FindMember(member.c_str());
691 if (mit != jvalue_->MemberEnd()) {
692 // FindMember returns a member iterator, but ConstDataIterator expects value iterator
693 // Since these are incompatible types, we return the value as a ConstDataIterator
694 // pointing to the member's value
695 return ConstDataIterator(jdoc_, &mit->value);
696 }
697 DN_CORE_TRACE("Member '{}' not found in object", member);
698 return ConstDataIterator();
699 } else {
700 DN_CORE_WARN("JSONValue is not an object, cannot find member!");
701 }
702 }
703 DN_CORE_WARN("JSONValue is empty!");
704 return ConstDataIterator();
705 }
