69bool duin::Script::Compile()
70{
71 DN_CORE_WARN("Compiling script {}...", scriptPath);
72 bool res = false;
73
74 // Set optional project file, script root, configure policies
75 if (!projectFile.empty())
76 {
77 fileAccess = das::make_smart<das::FsFileAccess>(projectFile, das::make_smart<das::FsFileAccess>());
78 }
79 else
80 {
81 fileAccess = das::make_smart<das::FsFileAccess>();
82 }
83 auto &fAccess = fileAccess;
84 fAccess->addFsRoot("scripts", "scripts");
85 das::CodeOfPolicies policies;
86 policies.rtti = true;
87
88 // Attempt compilation
89 // only update script program on successful compilation
90 auto compileStart = std::chrono::steady_clock::now();
91 das::ProgramPtr newProgram = das::compileDaScript(scriptPath, fAccess, tout, libGroup, policies);
92 auto compileMs =
93 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - compileStart).count();
94 if (newProgram->failed())
95 {
96 DN_CORE_FATAL("Compilation failed!");
97 for (auto &err : newProgram->errors)
98 {
99 DN_CORE_FATAL("{}", SafeErrorReport(err));
100 }
101
102 lastCompileError.clear();
103 for (auto &err : newProgram->errors)
104 {
105 lastCompileError += SafeErrorReport(err);
106 }
107
108 DN_CORE_ERROR("Compilation failed in {}ms.", compileMs);
109 res = false;
110 }
111 else
112 {
113 DN_CORE_INFO("Compiled {} in {}ms.", scriptPath, compileMs);
114 program = newProgram;
115 lastCompileError.clear();
116 res = true;
117 }
118
119 return res;
120}

173void duin::Script::ResetToBaseModules()
174{
175 // Order matters here:
176 // 1. libGroup.reset() first: skips promoted modules (builtIn=true, no delete) but clears
177 // the local vector — so their raw pointers are gone from libGroup before step 2.
178 // 2. Module::Reset(false) then: walks the global daScriptEnvironment::modules linked list
179 // and deletes only entries where promoted==true (i.e. daslib/*.das files promoted to
180 // builtins during a previous compileDaScript). This forces re-reading them from disk so
181 // hot-reload sees any changes. Duin's own C++ modules (builtIn=true, promoted=false)
182 // are never touched by Module::Reset and survive across recompiles unchanged.
183 // Doing this after libGroup.reset() avoids a double-free: if they are deleted first,
184 // libGroup.reset() would still dereference their (now-dangling) pointers.
185 libGroup.reset();
186 das::Module::Reset(false);
187 for (das::Module *m : baseModules)
188 {
189 libGroup.addModule(m);
190 }
191}

284std::string duin::Script::SafeErrorReport(const das::Error &err)
285{
286 std::ostringstream oss;
287
288 oss << "\n";
289 oss << "What: " << err.what << "\n";
290 oss << "Extra: " << err.extra << "\n";
291 oss << "Fixme: " << err.fixme << "\n";
292
293 // Guard against null fileInfo
294 if (err.at.fileInfo != nullptr)
295 {
296 try
297 {
298 oss << "At: " << err.at.describe() << "\n";
299 }
300 catch (...)
301 {
302 oss << "At: <describe() threw>\n";
303 }
304 }
305 else
306 {
307 oss << "At: " << err.at.line << ":" << err.at.column << " <no file>\n";
308 }
309
310 oss << "Cerr: " << (int)err.cerr << "\n";
311
312 return oss.str();
313}

315bool duin::VerifyFunction(das::Context *ctx, das::Func fn)
316{
317 if (!fn || !ctx)
318 return false;
319 // Pointer-range check: valid handles live inside the current context's
320 // contiguous function table. A stale SimFunction* from a destroyed context
321 // (Windows debug fill: 0xDDDDDDDD...) will fall outside this range without
322 // requiring us to dereference the potentially dangling pointer.
323 const das::SimFunction *begin = ctx->getFunction(0);
324 const das::SimFunction *end = begin + ctx->getTotalFunctions();
325 return fn.PTR >= begin && fn.PTR < end;
326}
