131void CameraMoveToTarget(CameraImpl &cameraImpl, float delta)
132{
133 Vector3 direction = Vector3Subtract(cameraImpl.target, cameraImpl.position);
134 float distance = Vector3LengthF(direction);
135
136 if (distance > 0.0f)
137 {
138 direction = Vector3Scale(Vector3NormalizeF(direction), fminf(delta, distance));
139 cameraImpl.position = Vector3Add(cameraImpl.position, direction);

143void CameraYaw(CameraImpl &cameraImpl, float angle, bool rotateAroundTarget)
144{
145 if (rotateAroundTarget)
146 {
147 Vector3 offset = Vector3Subtract(cameraImpl.position, cameraImpl.target);
148 Vector3 rotOffset = Vector3RotateByAxisAngle(offset, Vector3::UP, angle);
