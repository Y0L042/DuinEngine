154void CameraYaw(Camera *camera, float angle, bool rotateAroundTarget)
155{
156 if (camera == nullptr || !camera->IsValid())
157 {
158 DN_CORE_WARN("Camera pointer not valid!");
159 return;
160 }
161 Vector3 forward = GetCameraForward(camera);
162
163 if (rotateAroundTarget)
164 {
165 Vector3 offset = Vector3Subtract(camera->position, camera->target);
166 Vector3 rotOffset = Vector3RotateByAxisAngle(offset, Vector3::UP, angle);
167 camera->position = Vector3Add(camera->target, rotOffset);
168 camera->up = Vector3::UP;
169 }
170 else
171 {
172 // Normal Operation
173 Vector3 up = GetCameraUp(camera);
174 Matrix rotation = MatrixRotate(up, angle);
175 forward = Vector3Transform(forward, rotation);
176 camera->target = Vector3Add(camera->position, forward);
177 camera->up = Vector3Transform(camera->up, rotation);
178 }
179}

270void GetBGFXMatrix(Camera *camera)
271{
272 if (camera == nullptr || !camera->IsValid())
273 {
274 DN_CORE_WARN("Camera pointer not valid!");
275 return;
276 }
277
278 int WINDOW_WIDTH = GetWindowWidth();
279 int WINDOW_HEIGHT = GetWindowHeight();
280
281 Vector3 cTarget = camera->target;
282 Vector3 cPosition = camera->position;
283 Vector3 cUp = camera->up;
284 float fovy = camera->fovy;
285
286 const bx::Vec3 at = {cTarget.x, cTarget.y, cTarget.z};
287 const bx::Vec3 eye = {cPosition.x, cPosition.y, cPosition.z};
288 const bx::Vec3 up = {cUp.x, cUp.y, cUp.z};
289
290 float view[16];
291 bx::mtxLookAt(view, eye, at, up);
292
293 float proj[16];
294 bx::mtxProj(proj, fovy, float(WINDOW_WIDTH) / float(WINDOW_HEIGHT), 0.1f, 100.0f,
295 bgfx::getCaps()->homogeneousDepth);
296
297 bgfx::setViewTransform(0, view, proj);
298 bgfx::setViewRect(0, 0, 0, uint16_t(WINDOW_WIDTH), uint16_t(WINDOW_HEIGHT));
299}
