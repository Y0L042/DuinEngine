26 template <typename T>
27 void RegisterComponent(flecs::world &world, flecs::entity typeEntity)
28 {
29 std::string typeName = static_cast<std::string>(typeEntity.name());
30
31 if (typeAliases_.contains(typeName)) // && serializers_.contains(typeName) && deserializers_.contains(typeName))
32 {
33 return;
34 }
35
36 // Serialize function
37 serializers_[typeName] = [](const void *ptr) -> std::string {
38 if (ptr == nullptr) /* Tag */
39 {
40 const T tag;
41 return rfl::json::write<rfl::AddStructName<"type">>(tag);
42 }
43 else /* Component */
44 {
45 const T *typed_ptr = static_cast<const T *>(ptr);
46 return rfl::json::write<rfl::AddStructName<"type">>(*typed_ptr);
47 }
48 };
49
50 // Deserialize function
51 bool isTag = ecs_id_is_tag(world.c_ptr(), typeEntity.raw_id());
52 if (isTag)
53 {
54 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) { e.Add<T>(); };
55 }
56 else /* Component */
57 {
58 // Check if T has ReflectionType (uses the reflection pattern)
59 if constexpr (requires { typename T::ReflectionType; })
60 {
61 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) {
62 // Deserialize the ReflectionType (e.g., Transform3DImpl)
63 auto result = rfl::json::read<typename T::ReflectionType>(json);
64 if (result)
65 {
66 // Construct T from ReflectionType using T(const ReflectionType&) constructor
67 T component(result.value());
68 e.Set<T>(component);
69 }
70 };
71 }
72 else
73 {
74 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) {
75 T *typed_ptr = static_cast<T *>(ptr);
76 auto result = rfl::json::read<T>(json);
77 if (result)
78 {
79 typed_ptr = new T;
80 *typed_ptr = result.value();
81 e.Set<T>(*typed_ptr);
82 delete typed_ptr;
83 }
84 };
85 }
86 }
87
88 // Register type alias for types with ReflectionType (like Transform3D -> Transform3DImpl)
89 if constexpr (requires { typename T::ReflectionType; })
90 {
91 std::string fullReflectionTypeName = rfl::type_name_t<typename T::ReflectionType>().str();
92
93 // Extract unqualified name (e.g., "duin::ECSComponent::Velocity2D::Velocity2DImpl" -> "Velocity2DImpl")
94 std::string unqualifiedName = fullReflectionTypeName;
95 size_t lastColonPos = fullReflectionTypeName.find_last_of(':');
96 if (lastColonPos != std::string::npos)
97 {
98 unqualifiedName = fullReflectionTypeName.substr(lastColonPos + 1);
99 }
100
101 typeAliases_[unqualifiedName] = typeName;
102 DN_CORE_INFO("Registered type alias: {} -> {}", unqualifiedName, typeName);
103 }
104
105 DN_CORE_INFO("Registered {} {}", isTag ? "Tag" : "Component", typeName);
106 }

109 {
110 auto it = serializers_.find(typeName);
111 if (it != serializers_.end())
112 {
113 return it->second(componentPtr);
114 }
115 return "{}";
116 }

123 void Deserialize(Entity e, const std::string &typeName, void *componentPtr, const std::string &json)
124 {
125 // Check if this is an aliased type name (e.g., "Transform3DImpl" -> "Transform3D")
126 std::string resolvedTypeName = typeName;
127 auto aliasIt = typeAliases_.find(typeName);
128 if (aliasIt != typeAliases_.end())
129 {
130 resolvedTypeName = aliasIt->second;
131 }
132
133 auto it = deserializers_.find(resolvedTypeName);
134 if (it != deserializers_.end())
135 {
136 it->second(e, componentPtr, json);
137 }
138 }
