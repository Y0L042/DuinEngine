28 template <typename T>
29 void RegisterComponent(flecs::world &world, flecs::entity typeEntity)
30 {
31 std::string typeName = static_cast<std::string>(typeEntity.name());
32
33 if (typeAliases_.contains(typeName)) // && serializers_.contains(typeName) && deserializers_.contains(typeName))
34 {
35 return;
36 }
37
38 // Serialize function
39 serializers_[typeName] = [](const void *ptr) -> std::string {
40 if (ptr == nullptr) /* Tag */
41 {
42 const T tag;
43 return rfl::json::write<rfl::AddStructName<"type">>(tag);
44 }
45 else /* Component */
46 {
47 const T *typed_ptr = static_cast<const T *>(ptr);
48 return rfl::json::write<rfl::AddStructName<"type">>(*typed_ptr);
49 }
50 };
51
52 // PlainDeserialize function - simply adds component type, does not set data
53 plainDeserializers_[typeName] = [](Entity e, const std::string &type) { e.Add<T>(); };
54
55 remove_[typeName] = [](Entity e, const std::string &typeName) { e.Remove<T>(); };
56
57 // Deserialize function
58 bool isTag = ecs_id_is_tag(world.c_ptr(), typeEntity.raw_id());
59 if (isTag)
60 {
61 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) { e.Add<T>(); };
62 }
63 else /* Component */
64 {
65 // Check if T has ReflectionType (uses the reflection pattern)
66 if constexpr (requires { typename T::ReflectionType; })
67 {
68 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) {
69 // Deserialize the ReflectionType (e.g., Transform3DImpl)
70 auto result = rfl::json::read<typename T::ReflectionType>(json);
71 if (result)
72 {
73 // Construct T from ReflectionType using T(const ReflectionType&) constructor
74 T component(result.value());
75 e.Set<T>(component);
76 }
77 };
78 }
79 else
80 {
81 deserializers_[typeName] = [](Entity e, void *ptr, const std::string &json) {
82 T *typed_ptr = static_cast<T *>(ptr);
83 auto result = rfl::json::read<T>(json);
84 if (result)
85 {
86 typed_ptr = new T;
87 *typed_ptr = result.value();
88 e.Set<T>(*typed_ptr);
89 delete typed_ptr;
90 }
91 };
92 }
93 }
94
95 // Register type alias for types with ReflectionType (like Transform3D -> Transform3DImpl)
96 if constexpr (requires { typename T::ReflectionType; })
97 {
98 std::string fullReflectionTypeName = rfl::type_name_t<typename T::ReflectionType>().str();
99
100 // Extract unqualified name (e.g., "duin::ECSComponent::Velocity2D::Velocity2DImpl" -> "Velocity2DImpl")
101 std::string unqualifiedName = fullReflectionTypeName;
102 size_t lastColonPos = fullReflectionTypeName.find_last_of(':');
103 if (lastColonPos != std::string::npos)
104 {
105 unqualifiedName = fullReflectionTypeName.substr(lastColonPos + 1);
106 }
107
108 typeAliases_[unqualifiedName] = typeName;
109 // DN_CORE_INFO("Registered type alias: {} -> {}", unqualifiedName, typeName);
110 }
111
112 // DN_CORE_INFO("Registered {} {}", isTag ? "Tag" : "Component", typeName);
113 }
