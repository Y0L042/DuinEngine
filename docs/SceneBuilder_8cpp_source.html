224 PackedScene ps;

228 ps.uuid = UUID::FromStringHex(json.GetMember(ps.TAG_SCENEUUID).GetString());
229 }
230
231 if (json.HasMember(ps.TAG_SCENENAME))
232 {
233 ps.name = json.GetMember(ps.TAG_SCENENAME).GetString();
234 }
235
236 if (json.HasMember(ps.TAG_METADATA))

240 ps.metadata.engineVersion = metadataTemp.engineVersion;

242 ps.metadata.author = metadataTemp.author;
243 }

246 {
247 JSONValue depsArray = json.GetMember(ps.TAG_EXTERNALDEPENDENCIES);
248 if (depsArray.IsArray())
249 {
250 for (auto it = depsArray.Begin(); it != depsArray.End(); ++it)

252 ps.externalDependencies.push_back(PackedExternalDependency::Deserialize(it.GetValue()));
253 }

272duin::JSONValue duin::PackedScene::Serialize(const PackedScene &scene)
273{
274 JSONValue json;
275 json.SetObject();
276
277 json.AddMember(scene.TAG_SCENEUUID, UUID::ToStringHex(scene.uuid));
278 json.AddMember(scene.TAG_SCENENAME, scene.name);
279 json.AddMember(scene.TAG_METADATA, PackedSceneMetadata::Serialize(scene.metadata));
280
281 JSONValue depsArray;
282 depsArray.SetArray();
283 for (const auto &dep : scene.externalDependencies)
284 {
285 depsArray.PushBack(PackedExternalDependency::Serialize(dep));
286 }
287 json.AddMember(scene.TAG_EXTERNALDEPENDENCIES, depsArray);
288
289 JSONValue entitiesArray;
290 entitiesArray.SetArray();
291 for (const auto &entity : scene.entities)
292 {
293 entitiesArray.PushBack(PackedEntity::Serialize(entity));
294 }
295 json.AddMember(scene.TAG_ENTITIES, entitiesArray);
296
297 return json;
