123void GameWorld::SetGlobalTransform(duin::Entity e, ECSComponent::Transform3D tx)
124{
125 if (!e.IsValid() || !e.Has<ECSComponent::Transform3D>())
126 {
127 DN_CORE_WARN("Entity not valid, or does not have Transform3D!");
128 return;
129 }
130 ECSComponent::Transform3D *comp = e.TryGetMut<ECSComponent::Transform3D>();
131 if (!comp)
132 {
133 return;
134 }
135
136 duin::Entity parent = e.Parent();
137 if (parent.IsValid() && parent.Has<ECSComponent::Transform3D>())
138 {
139 ECSComponent::Transform3D parentGlobal = GetGlobalTransform(parent);
140
141 Vector3 offset = Vector3Subtract(tx.GetPosition(), parentGlobal.GetPosition());
142 Quaternion invParentRot = QuaternionInvert(parentGlobal.GetRotation());
143 Vector3 localPosUnscaled = Vector3RotateByQuaternion(offset, invParentRot);
144 Vector3 localPos = Vector3Divide(localPosUnscaled, parentGlobal.GetScale());
145
146 Vector3 localScale = Vector3Divide(tx.GetScale(), parentGlobal.GetScale());
147
148 Quaternion localRot = QuaternionMultiply(QuaternionInvert(parentGlobal.GetRotation()), tx.GetRotation());
149
150 comp->SetPosition(localPos);
151 comp->SetScale(localScale);
152 comp->SetRotation(localRot);
153 }
154 else
155 {
156 comp->SetPosition(tx.GetPosition());
157 comp->SetScale(tx.GetScale());
158 comp->SetRotation(tx.GetRotation());
159 }
160 comp->UpdateGlobalPositionCache(tx.GetPosition());
161 comp->UpdateGlobalScaleCache(tx.GetScale());
162 comp->UpdateGlobalRotationCache(tx.GetRotation());
163}

165ECSComponent::Transform3D GameWorld::GetGlobalTransform(duin::Entity e)
166{
167 if (!e.IsValid() || !e.Has<ECSComponent::Transform3D>())
168 {
169 return ECSComponent::Transform3D();
170 }
171 ECSComponent::Transform3D *tx = e.TryGetMut<ECSComponent::Transform3D>();
172 if (!tx)
173 {
174 return ECSComponent::Transform3D();
175 }
176
177 duin::Entity parent = e.Parent();
178 if (parent.IsValid() && parent.Has<ECSComponent::Transform3D>())
179 {
180 ECSComponent::Transform3D parentGlobal = GetGlobalTransform(parent);
181
182 Vector3 scaledLocalPos = Vector3Multiply(tx->GetPosition(), parentGlobal.GetScale());
183 Vector3 rotatedPos = Vector3RotateByQuaternion(scaledLocalPos, parentGlobal.GetRotation());
184 Vector3 globalPos = Vector3Add(parentGlobal.GetPosition(), rotatedPos);
185
186 Vector3 globalScale = Vector3Multiply(parentGlobal.GetScale(), tx->GetScale());
187
188 Quaternion globalRot = QuaternionMultiply(parentGlobal.GetRotation(), tx->GetRotation());
189
190 tx->UpdateGlobalPositionCache(globalPos);
191 tx->UpdateGlobalScaleCache(globalScale);
192 tx->UpdateGlobalRotationCache(globalRot);
193
194 return ECSComponent::Transform3D(globalPos, globalScale, globalRot);
195 }
196 else
197 {
198 tx->UpdateGlobalPositionCache(tx->GetPosition());
199 tx->UpdateGlobalScaleCache(tx->GetScale());
200 tx->UpdateGlobalRotationCache(tx->GetRotation());
201
202 return *tx;
203 }

236Vector3 GameWorld::GetGlobalPosition(duin::Entity e)
237{
238 if (!e.IsValid() || !e.Has<ECSComponent::Transform3D>())
239 {
240 //DN_CORE_WARN("Entity not valid, or does not have Transform3D!");

248
249 if (tx->globalPositionCacheDirtyFlag)

253 {
254 tx->UpdateGlobalPositionCache(tx->GetPosition());
255 }
256 else
257 {
258 Vector3 parentGlobalPos = GetGlobalPosition(parent);
259 Quaternion parentGlobalRot = GetGlobalRotation(parent);
260 Vector3 parentGlobalScale = GetGlobalScale(parent);
261 Vector3 scaledLocalPos = Vector3Multiply(tx->GetPosition(), parentGlobalScale);
262 Vector3 rotatedPos = Vector3RotateByQuaternion(scaledLocalPos, parentGlobalRot);
263 Vector3 globalPos = Vector3Add(parentGlobalPos, rotatedPos);
264 tx->UpdateGlobalPositionCache(globalPos);
265 }
