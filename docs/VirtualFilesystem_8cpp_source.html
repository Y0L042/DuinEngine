103std::string OverridePathPrefix(const std::string &path, const std::string &prefix)
104{
105 static constexpr size_t vpathSize = 6; // "xxx://" = 6 characters
106
107 if (path.size() < vpathSize || !fs::IsVirtualPath(path))
108 {
109 DN_WARN("OverridePathPrefix: path '{}' is not a valid virtual path!", path);
110 return std::string();
111 }
112
113 if (prefix.size() > vpathSize)
114 {
115 DN_WARN("OverridePathPrefix: prefix '{}' exceeds maximum length of {} characters!", prefix, vpathSize);
116 }
117
118 if (!fs::IsVirtualPath(prefix + path.substr(vpathSize)))
119 {
120 DN_WARN("OverridePathPrefix: prefix '{}' is not a recognized virtual prefix!", prefix);
121 return std::string();
122 }
123
124 return prefix + path.substr(vpathSize);
125}
