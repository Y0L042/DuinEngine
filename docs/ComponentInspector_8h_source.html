37 // InspectFn receives:
38 // ptr - mutable raw pointer to the live component data (may be null for tags)
39 // e - the entity owning the component (for Set<T> writeback)
40 using InspectFn = std::function<void(void *ptr, Entity &e)>;
41
47 template <typename T>
48 void RegisterComponent(const std::string &typeName)
49 {
50 if (inspectors_.contains(typeName))
51 return;
52
53 if constexpr (std::is_empty_v<T>)
54 {
55 // Tags have no data
56 inspectors_[typeName] = [](void *, Entity &) { ImGui::TextDisabled("(tag)"); };
57 }
58 else if constexpr (requires { typename T::ReflectionType; })
59 {
60 // Component with ReflectionType pattern:
61 // - read fields from a copy of reflection()
62 // - write back via T(edited_impl) + entity.Set<T>
63 inspectors_[typeName] = [](void *ptr, Entity &e) {
64 if (!ptr)
65 {
66 ImGui::TextDisabled("(null ptr)");
67 return;
68 }
69 T *comp = static_cast<T *>(ptr);
70 typename T::ReflectionType impl = comp->reflection();
71 bool changed = false;
72 auto view = rfl::to_view(impl);
73 view.apply([&changed](const auto &field) {
74 if (DrawField(field.name(), *field.value()))
75 changed = true;
76 });
77 if (changed)
78 e.Set<T>(T(impl));
79 };
80 }
81 else if constexpr (std::is_aggregate_v<T> && std::is_class_v<T>)
82 {
83 // Aggregate struct without ReflectionType: use rfl::to_view directly.
84 // Changes are written back in-place since ptr is already the live data.
85 inspectors_[typeName] = [](void *ptr, Entity &) {
86 if (!ptr)
87 {
88 ImGui::TextDisabled("(null ptr)");
89 return;
90 }
91 T *comp = static_cast<T *>(ptr);
92 auto view = rfl::to_view(*comp);
93 view.apply([](const auto &field) { DrawField(field.name(), *field.value()); });
94 };
95 }
96 else
97 {
98 // Not reflectable and no ReflectionType — show opaque placeholder
99 inspectors_[typeName] = [](void *, Entity &) { ImGui::TextDisabled("(opaque)"); };
100 }
101 }
102
110 bool Inspect(const std::string &typeName, void *ptr, Entity &e) const
111 {
112 auto it = inspectors_.find(typeName);
113 if (it == inspectors_.end())
114 return false;
115 it->second(ptr, e);
116 return true;
117 }
118
119 bool IsRegistered(const std::string &typeName) const
120 {
121 return inspectors_.contains(typeName);
122 }

131 template <typename T>
132 static bool DrawField(std::string_view name, T &value)
133 {
134 // Build a unique ImGui label: "fieldName##addr"
135 char label[128];
136 std::snprintf(label, sizeof(label), "%.*s", static_cast<int>(name.size()), name.data());
137
138 if constexpr (std::is_same_v<T, float>)
139 {
140 return ImGui::DragFloat(label, &value, 0.05f);
141 }
142 else if constexpr (std::is_same_v<T, double>)
143 {
144 float v = static_cast<float>(value);
145 if (ImGui::DragFloat(label, &v, 0.05f))
146 {
147 value = static_cast<double>(v);
148 return true;
149 }
150 return false;
151 }
152 else if constexpr (std::is_same_v<T, int> || std::is_same_v<T, int32_t>)
153 {
154 return ImGui::DragInt(label, &value);
155 }
156 else if constexpr (std::is_same_v<T, uint32_t>)
157 {
158 int v = static_cast<int>(value);
159 if (ImGui::DragInt(label, &v, 1, 0, INT_MAX))
160 {
161 value = static_cast<uint32_t>(v);
162 return true;
163 }
164 return false;
165 }
166 else if constexpr (std::is_same_v<T, bool>)
167 {
168 return ImGui::Checkbox(label, &value);
169 }
170 else if constexpr (std::is_same_v<T, std::string>)
171 {
172 char buf[256];
173 std::strncpy(buf, value.c_str(), sizeof(buf) - 1);
174 buf[sizeof(buf) - 1] = '\0';
175 if (ImGui::InputText(label, buf, sizeof(buf)))
176 {
177 value = buf;
178 return true;
179 }
180 return false;
181 }
182 else if constexpr (std::is_pointer_v<T> || requires { typename T::element_type; })
183 {
184 // Pointers / smart pointers — show read-only address
185 ImGui::LabelText(label, "%p", static_cast<const void *>(&value));
186 return false;
187 }
188 else if constexpr (requires { typename T::ReflectionType; })
189 {
190 // Nested type with ReflectionType — recurse via reflection()
191 bool changed = false;
192 if (ImGui::TreeNodeEx(label, ImGuiTreeNodeFlags_DefaultOpen))
193 {
194 typename T::ReflectionType impl = value.reflection();
195 auto view = rfl::to_view(impl);
196 view.apply([&changed](const auto &f) {
197 if (DrawField(f.name(), *f.value()))
198 changed = true;
199 });
200 if (changed)
201 value = T(impl);
202 ImGui::TreePop();
203 }
204 return changed;
205 }
206 else if constexpr (std::is_aggregate_v<T> && std::is_class_v<T>)
207 {
208 // Generic aggregate struct — recurse via rfl::to_view directly
209 bool changed = false;
210 if (ImGui::TreeNodeEx(label, ImGuiTreeNodeFlags_DefaultOpen))
211 {
212 auto view = rfl::to_view(value);
213 view.apply([&changed](const auto &f) {
214 if (DrawField(f.name(), *f.value()))
215 changed = true;
216 });
217 ImGui::TreePop();
218 }
219 return changed;
220 }
221 else
222 {
223 // Scalar or opaque type — show read-only placeholder
224 ImGui::LabelText(label, "(opaque)");
225 return false;
226 }
227 }
