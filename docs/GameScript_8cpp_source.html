78bool duin::GameScript::CompileAndSimulate()
79{
80 bool isReload = hasCompiledOnce;
81
82 ResetMuteWarningFlags();
83 ResetToBaseModules();
84
85 bool compiled = Compile();
86 if (compiled)
87 {
88 // Call pre-Reload hooks of old context
89 if (isReload && scriptReady)
90 {
91 g_DnLiveHostState.is_reload = true;
92 for (auto *fn : g_DnLiveHostState.beforeReloadFns)
93 {
94 CallScript(fn);
95 }
96 CallLiveVarsFunctions("__before_reload_live_vars");
97 }
98
99 g_DnLiveHostState.beforeReloadFns.clear();
100 g_DnLiveHostState.afterReloadFns.clear();
101
102 std::shared_ptr<ScriptMemory> oldContextMemory;
103 if (context)
104 {
105 oldContextMemory = context->scriptMemory;
106 }
107
108 bool simulated = SimulateContext();
109 if (simulated)
110 {
111 if (oldContextMemory)
112 {
113 context->scriptMemory = oldContextMemory;
114 }
115
116 SetGameFunctions();
117 SetContextRootObject();
118 g_DnLiveHostState.last_error.clear();
119
120 if (isReload)
121 {
122 for (auto *fn : g_DnLiveHostState.afterReloadFns)
123 {
124 CallScript(fn);
125 }
126 CallLiveVarsFunctions("__after_reload_live_vars");
127 g_DnLiveHostState.is_reload = false;
128 }
129
130 if (!hotCompileEnabled)
131 {
132 // Treat recompile as a fresh start: clear only once we have a working new script.
133 ClearScriptGameObjects();
134 }
135
136 hasCompiledOnce = true;
137 Ready();
138 }
139 else
140 {
141 g_DnLiveHostState.last_error = "Simulation failed";
142 g_DnLiveHostState.is_reload = false;
143 }
144
145 compiled = simulated;
146 }
147 else
148 {
149 g_DnLiveHostState.last_error = lastCompileError;
150
151 if (haltOnCompilationFail)
152 {
153 scriptReady = false;
154 }
155 }
156
157 return compiled;
158}

242void duin::GameScript::Update(double delta)
243{
244 uptimeAccum += static_cast<float>(delta);
245 g_DnLiveHostState.dt = static_cast<float>(delta);
246 g_DnLiveHostState.uptime = uptimeAccum;
247 g_DnLiveHostState.fps = (delta > 0.0) ? static_cast<float>(1.0 / delta) : 0.0f;
248
249 // Only hot recompile if it has been X seconds since last file change (with no further changes).
250 // Timer then disabled to prevent continual recompilation, only reset when file is changed again.
251 float cooldownTimerPrev = hotCompileFileChangeCooldownTimer;
252 if (hotCompileFileChangeCooldownTimer > 0.0f)
253 {
254 hotCompileFileChangeCooldownTimer -= delta;
255 }
256 bool cooldownReachedZero =
257 (hotCompileFileChangeCooldownTimer <= 0.0f) && ((hotCompileFileChangeCooldownTimer * cooldownTimerPrev) < 0.0f);
258 if (cooldownReachedZero)
259 {
260 queueHotCompileFlag = true;
261 hotCompileFileChangeCooldownTimer = -999.999f;
262 }
263
264 if (queueHotCompileFlag)
265 {
266 queueHotCompileFlag = false;
267 if (hotCompileEnabled)
268 {
269 duin::fs::PathInfo pInfo;
270 if (duin::vfs::GetPathInfo("bin://" + scriptPath, &pInfo))
271 {
272 scriptLastModified = pInfo.modifyTime;
273 }
274 }
275 if (!CompileAndSimulate())
276 {
277 DN_CORE_WARN("Compilation of {} failed! Continuing with previous version.", "bin://" + scriptPath);
278 }
279 }
280
281 if (scriptReady)
282 {
283 vec4f args[1];
284 args[0] = cast<double>::from(delta);
285 CallScript(fnGameUpdate, args);
286 }
287 else
288 {
289 if (!muteUpdateWarning)
290 {
291 DN_CORE_WARN("(Update) Script not ready! {}", scriptPath);
292 muteUpdateWarning = true;
293 }
294 }
295}
