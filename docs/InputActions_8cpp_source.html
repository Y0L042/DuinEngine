28void duin::AddInputActionBinding(const std::string &actionName, const InputBinding &newBinding)
29{
30 if (inputActionMap.find(actionName) == inputActionMap.end())
31 {
32 // Create new Action
33 DN_CORE_WARN("InputAction {} not found, creating new Action!", actionName);
34 if (!inputActionMap.try_emplace(actionName, InputAction()).second)
35 {
36 DN_CORE_WARN("Unable to create InputAction {}!", actionName);
37 return;
38 }
39 }
40
41 // Add binding
42 auto &action = inputActionMap[actionName];
43 for (auto &binding : action.inputBindings)
44 {
45 if (newBinding.bindingHash == binding.bindingHash)
46 {
47 DN_CORE_WARN("Binding for Action already exist!", actionName);
48 return;
49 }
50 }
51 action.inputBindings.push_back(newBinding);

60void duin::RemoveInputActionBinding(const std::string &actionName, const InputBinding &binding)
61{
62 if (inputActionMap.find(actionName) == inputActionMap.end())
63 {
64 DN_CORE_WARN("InputAction {} not found, unable to remove!", actionName);
65 return;
66 }
67 auto &action = inputActionMap[actionName];
68 auto it = action.inputBindings.begin();
69 while (it != action.inputBindings.end())
70 {
71 if (it->bindingHash == binding.bindingHash)
72 {
73 std::swap(*it, action.inputBindings.back());
74 action.inputBindings.pop_back();
75 break;
76 }
77 }
