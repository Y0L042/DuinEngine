50 return 1;

133 void update(float _deltaTime, const entry::MouseState& _mouseState, bool _reset)
134 {
135 if (_reset)
136 {
137 m_mouseLast.m_mx = _mouseState.m_mx;
138 m_mouseLast.m_my = _mouseState.m_my;
139 m_mouseLast.m_mz = _mouseState.m_mz;
140 m_mouseNow = m_mouseLast;
141 m_mouseDown = false;
142
143 return;
144 }
145
146 if (!m_mouseDown)
147 {
148 m_mouseLast.m_mx = _mouseState.m_mx;
149 m_mouseLast.m_my = _mouseState.m_my;
150 }
151
152 m_mouseDown = !!_mouseState.m_buttons[entry::MouseButton::Right];
153
154 if (m_mouseDown)
155 {
156 m_mouseNow.m_mx = _mouseState.m_mx;
157 m_mouseNow.m_my = _mouseState.m_my;
158 }
159
160 m_mouseLast.m_mz = m_mouseNow.m_mz;
161 m_mouseNow.m_mz = _mouseState.m_mz;
162
163 const float deltaZ = float(m_mouseNow.m_mz - m_mouseLast.m_mz);
164
165 if (m_mouseDown)
166 {
167 const int32_t deltaX = m_mouseNow.m_mx - m_mouseLast.m_mx;
168 const int32_t deltaY = m_mouseNow.m_my - m_mouseLast.m_my;
169
170 m_horizontalAngle += m_mouseSpeed * float(deltaX);
171 m_verticalAngle -= m_mouseSpeed * float(deltaY);
172
173 m_mouseLast.m_mx = m_mouseNow.m_mx;
174 m_mouseLast.m_my = m_mouseNow.m_my;
175 }
176
177 entry::GamepadHandle handle = { 0 };
178 m_horizontalAngle += m_gamepadSpeed * inputGetGamepadAxis(handle, entry::GamepadAxis::RightX)/32768.0f;
179 m_verticalAngle -= m_gamepadSpeed * inputGetGamepadAxis(handle, entry::GamepadAxis::RightY)/32768.0f;
180 const int32_t gpx = inputGetGamepadAxis(handle, entry::GamepadAxis::LeftX);
181 const int32_t gpy = inputGetGamepadAxis(handle, entry::GamepadAxis::LeftY);
182 m_keys |= gpx < -16834 ? CAMERA_KEY_LEFT : 0;
183 m_keys |= gpx > 16834 ? CAMERA_KEY_RIGHT : 0;
184 m_keys |= gpy < -16834 ? CAMERA_KEY_FORWARD : 0;
185 m_keys |= gpy > 16834 ? CAMERA_KEY_BACKWARD : 0;
186
187 const bx::Vec3 direction =
188 {
189 bx::cos(m_verticalAngle) * bx::sin(m_horizontalAngle),
190 bx::sin(m_verticalAngle),
191 bx::cos(m_verticalAngle) * bx::cos(m_horizontalAngle),
192 };
193
194 const bx::Vec3 right =
195 {
196 bx::sin(m_horizontalAngle - bx::kPiHalf),
197 0.0f,
198 bx::cos(m_horizontalAngle - bx::kPiHalf),
199 };
200
201 const bx::Vec3 up = bx::cross(right, direction);
202
203 m_eye = bx::mad(direction, deltaZ * _deltaTime * m_moveSpeed, m_eye);
204
205 if (m_keys & CAMERA_KEY_FORWARD)
206 {
207 m_eye = bx::mad(direction, _deltaTime * m_moveSpeed, m_eye);
208 setKeyState(CAMERA_KEY_FORWARD, false);
209 }
210
211 if (m_keys & CAMERA_KEY_BACKWARD)
212 {
213 m_eye = bx::mad(direction, -_deltaTime * m_moveSpeed, m_eye);
214 setKeyState(CAMERA_KEY_BACKWARD, false);
215 }
216
217 if (m_keys & CAMERA_KEY_LEFT)
218 {
219 m_eye = bx::mad(right, _deltaTime * m_moveSpeed, m_eye);
220 setKeyState(CAMERA_KEY_LEFT, false);
221 }
222
223 if (m_keys & CAMERA_KEY_RIGHT)
224 {
225 m_eye = bx::mad(right, -_deltaTime * m_moveSpeed, m_eye);
226 setKeyState(CAMERA_KEY_RIGHT, false);
227 }
228
229 if (m_keys & CAMERA_KEY_UP)
230 {
231 m_eye = bx::mad(up, _deltaTime * m_moveSpeed, m_eye);
232 setKeyState(CAMERA_KEY_UP, false);
233 }
234
235 if (m_keys & CAMERA_KEY_DOWN)
236 {
237 m_eye = bx::mad(up, -_deltaTime * m_moveSpeed, m_eye);
238 setKeyState(CAMERA_KEY_DOWN, false);
239 }
240
241 m_at = bx::add(m_eye, direction);
242 m_up = bx::cross(right, direction);
243 }
