31// This is the main rendering function that you have to implement and call after
32// ImGui::Render(). Pass ImGui::GetDrawData() to this function.
33// Note: If text or lines are blurry when integrating ImGui into your engine,
34// in your Render function, try translating your projection matrix by
35// (0.5f,0.5f) or (0.375f,0.375f)
36void ImGui_Implbgfx_RenderDrawLists(ImDrawData *draw_data)
37{
38 // Avoid rendering when minimized, scale coordinates for retina displays
39 // (screen coordinates != framebuffer coordinates)
40 ImGuiIO &io = ImGui::GetIO();
41 int fb_width = (int)(io.DisplaySize.x * io.DisplayFramebufferScale.x);
42 int fb_height = (int)(io.DisplaySize.y * io.DisplayFramebufferScale.y);
43 if (fb_width == 0 || fb_height == 0)
44 {
45 return;
46 }
47
48 draw_data->ScaleClipRects(io.DisplayFramebufferScale);
49
50 // Setup render state: alpha-blending enabled, no face culling,
51 // no depth testing, scissor enabled
52 uint64_t state = BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_MSAA |
53 BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA);
54
55 const bgfx::Caps *caps = bgfx::getCaps();
56
57 // Setup viewport, orthographic projection matrix
58 float ortho[16];
59 bx::mtxOrtho(ortho, 0.0f, io.DisplaySize.x, io.DisplaySize.y, 0.0f, 0.0f, 1000.0f, 0.0f, caps->homogeneousDepth);
60 bgfx::setViewTransform(g_View, NULL, ortho);
61 bgfx::setViewRect(g_View, 0, 0, (uint16_t)fb_width, (uint16_t)fb_height);
62
63 // Render command lists
64 for (int n = 0; n < draw_data->CmdListsCount; n++)
65 {
66 const ImDrawList *cmd_list = draw_data->CmdLists[n];
67
68 bgfx::TransientVertexBuffer tvb;
69 bgfx::TransientIndexBuffer tib;
70
71 uint32_t numVertices = (uint32_t)cmd_list->VtxBuffer.size();
72 uint32_t numIndices = (uint32_t)cmd_list->IdxBuffer.size();
73
74 if ((numVertices != bgfx::getAvailTransientVertexBuffer(numVertices, g_VertexLayout)) ||
75 (numIndices != bgfx::getAvailTransientIndexBuffer(numIndices)))
76 {
77 // not enough space in transient buffer, quit drawing the rest...
78 break;
79 }
80
81 bgfx::allocTransientVertexBuffer(&tvb, numVertices, g_VertexLayout);
82 bgfx::allocTransientIndexBuffer(&tib, numIndices);
83
84 ImDrawVert *verts = (ImDrawVert *)tvb.data;
85 memcpy(verts, cmd_list->VtxBuffer.begin(), numVertices * sizeof(ImDrawVert));
86
87 ImDrawIdx *indices = (ImDrawIdx *)tib.data;
88 memcpy(indices, cmd_list->IdxBuffer.begin(), numIndices * sizeof(ImDrawIdx));
89
90 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
91 {
92 const ImDrawCmd *pcmd = &cmd_list->CmdBuffer[cmd_i];
93
94 if (pcmd->UserCallback)
95 {
96 pcmd->UserCallback(cmd_list, pcmd);
97 }
98 else
99 {
100 const uint16_t xx = (uint16_t)bx::max(pcmd->ClipRect.x, 0.0f);
101 const uint16_t yy = (uint16_t)bx::max(pcmd->ClipRect.y, 0.0f);
102 bgfx::setScissor(xx, yy, (uint16_t)bx::min(pcmd->ClipRect.z, 65535.0f) - xx,
103 (uint16_t)bx::min(pcmd->ClipRect.w, 65535.0f) - yy);
104
105 bgfx::setState(state);
106 bgfx::TextureHandle texture = {(uint16_t)((intptr_t)pcmd->TextureId & 0xffff)};
107 bgfx::setTexture(0, g_AttribLocationTex, texture);
108 bgfx::setVertexBuffer(0, &tvb, 0, numVertices);
109 bgfx::setIndexBuffer(&tib, pcmd->IdxOffset, pcmd->ElemCount);
110 bgfx::submit(g_View, g_ShaderHandle);
111 }
112 }
113 }
114}

116bool ImGui_Implbgfx_CreateFontsTexture()
117{
118 // Build texture atlas
119 ImGuiIO &io = ImGui::GetIO();
120 unsigned char *pixels;
121 int width, height;
122 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
123
124 // Upload texture to graphics system
125 g_FontTexture = bgfx::createTexture2D((uint16_t)width, (uint16_t)height, false, 1, bgfx::TextureFormat::BGRA8, 0,
126 bgfx::copy(pixels, width * height * 4));
127
128 // Store our identifier TODO
129 io.Fonts->TexID = (ImTextureID)g_FontTexture.idx;
130
131 return true;
132}

140bool ImGui_Implbgfx_CreateDeviceObjects()
141{
142 bgfx::RendererType::Enum type = bgfx::getRendererType();
143 g_ShaderHandle = bgfx::createProgram(bgfx::createEmbeddedShader(s_embeddedShaders, type, "vs_ocornut_imgui"),
144 bgfx::createEmbeddedShader(s_embeddedShaders, type, "fs_ocornut_imgui"), true);
145
146 g_VertexLayout.begin()
147 .add(bgfx::Attrib::Position, 2, bgfx::AttribType::Float)
148 .add(bgfx::Attrib::TexCoord0, 2, bgfx::AttribType::Float)
149 .add(bgfx::Attrib::Color0, 4, bgfx::AttribType::Uint8, true)
150 .end();
151
152 g_AttribLocationTex = bgfx::createUniform("g_AttribLocationTex", bgfx::UniformType::Sampler);
153
154 ImGui_Implbgfx_CreateFontsTexture();
155
156 return true;
157}

159void ImGui_Implbgfx_InvalidateDeviceObjects()
160{

162 bgfx::destroy(g_ShaderHandle);
163
164 if (isValid(g_FontTexture))
165 {
166 bgfx::destroy(g_FontTexture);
167 ImGui::GetIO().Fonts->TexID = 0;
168 g_FontTexture.idx = bgfx::kInvalidHandle;
169 }
