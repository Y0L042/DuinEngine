31// This is the main rendering function that you have to implement and call after
32// ImGui::Render(). Pass ImGui::GetDrawData() to this function.
33// Note: If text or lines are blurry when integrating ImGui into your engine,
34// in your Render function, try translating your projection matrix by
35// (0.5f,0.5f) or (0.375f,0.375f)
36void ImGui_Implbgfx_RenderDrawLists(ImDrawData* draw_data)
37{
38 // Avoid rendering when minimized, scale coordinates for retina displays
39 // (screen coordinates != framebuffer coordinates)
40 ImGuiIO& io = ImGui::GetIO();
41 int fb_width = (int)(io.DisplaySize.x * io.DisplayFramebufferScale.x);
42 int fb_height = (int)(io.DisplaySize.y * io.DisplayFramebufferScale.y);
43 if (fb_width == 0 || fb_height == 0) {
44 return;
45 }
46
47 draw_data->ScaleClipRects(io.DisplayFramebufferScale);
48
49 // Setup render state: alpha-blending enabled, no face culling,
50 // no depth testing, scissor enabled
51 uint64_t state =
52 BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_MSAA |
53 BGFX_STATE_BLEND_FUNC(
54 BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA);
55
56 const bgfx::Caps* caps = bgfx::getCaps();
57
58 // Setup viewport, orthographic projection matrix
59 float ortho[16];
60 bx::mtxOrtho(
61 ortho, 0.0f, io.DisplaySize.x, io.DisplaySize.y, 0.0f, 0.0f, 1000.0f,
62 0.0f, caps->homogeneousDepth);
63 bgfx::setViewTransform(g_View, NULL, ortho);
64 bgfx::setViewRect(g_View, 0, 0, (uint16_t)fb_width, (uint16_t)fb_height);
65
66 // Render command lists
67 for (int n = 0; n < draw_data->CmdListsCount; n++) {
68 const ImDrawList* cmd_list = draw_data->CmdLists[n];
69
70 bgfx::TransientVertexBuffer tvb;
71 bgfx::TransientIndexBuffer tib;
72
73 uint32_t numVertices = (uint32_t)cmd_list->VtxBuffer.size();
74 uint32_t numIndices = (uint32_t)cmd_list->IdxBuffer.size();
75
76 if ((numVertices != bgfx::getAvailTransientVertexBuffer(
77 numVertices, g_VertexLayout)) ||
78 (numIndices != bgfx::getAvailTransientIndexBuffer(numIndices))) {
79 // not enough space in transient buffer, quit drawing the rest...
80 break;
81 }
82
83 bgfx::allocTransientVertexBuffer(&tvb, numVertices, g_VertexLayout);
84 bgfx::allocTransientIndexBuffer(&tib, numIndices);
85
86 ImDrawVert* verts = (ImDrawVert*)tvb.data;
87 memcpy(
88 verts, cmd_list->VtxBuffer.begin(),
89 numVertices * sizeof(ImDrawVert));
90
91 ImDrawIdx* indices = (ImDrawIdx*)tib.data;
92 memcpy(
93 indices, cmd_list->IdxBuffer.begin(),
94 numIndices * sizeof(ImDrawIdx));
95
96 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++) {
97 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
98
99 if (pcmd->UserCallback) {
100 pcmd->UserCallback(cmd_list, pcmd);
101 } else {
102 const uint16_t xx = (uint16_t)bx::max(pcmd->ClipRect.x, 0.0f);
103 const uint16_t yy = (uint16_t)bx::max(pcmd->ClipRect.y, 0.0f);
104 bgfx::setScissor(
105 xx, yy, (uint16_t)bx::min(pcmd->ClipRect.z, 65535.0f) - xx,
106 (uint16_t)bx::min(pcmd->ClipRect.w, 65535.0f) - yy);
107
108 bgfx::setState(state);
109 bgfx::TextureHandle texture = {
110 (uint16_t)((intptr_t)pcmd->TextureId & 0xffff)};
111 bgfx::setTexture(0, g_AttribLocationTex, texture);
112 bgfx::setVertexBuffer(0, &tvb, 0, numVertices);
113 bgfx::setIndexBuffer(&tib, pcmd->IdxOffset, pcmd->ElemCount);
114 bgfx::submit(g_View, g_ShaderHandle);
115 }
116 }
117 }
118}

120bool ImGui_Implbgfx_CreateFontsTexture()
121{
122 // Build texture atlas
123 ImGuiIO& io = ImGui::GetIO();
124 unsigned char* pixels;
125 int width, height;
126 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
127
128 // Upload texture to graphics system
129 g_FontTexture = bgfx::createTexture2D(
130 (uint16_t)width, (uint16_t)height, false, 1, bgfx::TextureFormat::BGRA8,
131 0, bgfx::copy(pixels, width * height * 4));
132
133 // Store our identifier TODO
134 io.Fonts->TexID = (ImTextureID)g_FontTexture.idx;
135
136 return true;
137}

146bool ImGui_Implbgfx_CreateDeviceObjects()
147{
148 bgfx::RendererType::Enum type = bgfx::getRendererType();
149 g_ShaderHandle = bgfx::createProgram(
150 bgfx::createEmbeddedShader(s_embeddedShaders, type, "vs_ocornut_imgui"),
151 bgfx::createEmbeddedShader(s_embeddedShaders, type, "fs_ocornut_imgui"),
152 true);
153
154 g_VertexLayout.begin()
155 .add(bgfx::Attrib::Position, 2, bgfx::AttribType::Float)
156 .add(bgfx::Attrib::TexCoord0, 2, bgfx::AttribType::Float)
157 .add(bgfx::Attrib::Color0, 4, bgfx::AttribType::Uint8, true)
158 .end();
159
160 g_AttribLocationTex =
161 bgfx::createUniform("g_AttribLocationTex", bgfx::UniformType::Sampler);
162
163 ImGui_Implbgfx_CreateFontsTexture();
164
165 return true;
166}

168void ImGui_Implbgfx_InvalidateDeviceObjects()
169{
170 bgfx::destroy(g_AttribLocationTex);
171 bgfx::destroy(g_ShaderHandle);
172
173 if (isValid(g_FontTexture)) {
174 bgfx::destroy(g_FontTexture);
175 ImGui::GetIO().Fonts->TexID = 0;
176 g_FontTexture.idx = bgfx::kInvalidHandle;
177 }
178}
