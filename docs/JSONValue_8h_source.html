465 template <typename T>
466 JSONValue& AddMember(const std::string& key, T val, bool allowDuplicates = false) // Add Key:Value member to object
467 {
468 if (!jvalue_->IsObject()) {
469 DN_CORE_WARN("JSONValue not an Object, cannot add member!");
470 return *this;
471 }
472
473 if (!allowDuplicates) {
474 RemoveMember(key.c_str());
475 }
476
477 rapidjson::Value name(key.c_str(),
478 static_cast<rapidjson::SizeType>(key.size()),
479 jdoc_->GetAllocator());
480
481 rapidjson::Value v;
482 if constexpr (std::is_same_v<T, std::string>) {
483 v.SetString(val.c_str(),
484 static_cast<rapidjson::SizeType>(val.size()),
485 jdoc_->GetAllocator());
486 }
487 else if constexpr (std::is_same_v<T, char>) {
488 char buf[2] = { val, '\0' };
489 v.SetString(buf,
490 static_cast<rapidjson::SizeType>(sizeof(char)),
491 jdoc_->GetAllocator());
492
493 }
494 else {
495 v.Set(val);
496 }
497
498 jvalue_->AddMember(std::move(name),
499 std::move(v),
500 jdoc_->GetAllocator());
501
502 return *this;
503 }
