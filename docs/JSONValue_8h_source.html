466 template <typename T>
467 JSONValue &AddMember(const std::string &key, T val, bool allowDuplicates = false) // Add Key:Value member to object
468 {
469 if (!jvalue_->IsObject())
470 {
471 DN_CORE_WARN("JSONValue not an Object, cannot add member!");
472 return *this;
473 }
474
475 if (!allowDuplicates)
476 {
477 RemoveMember(key.c_str());
478 }
479
480 rapidjson::Value name(key.c_str(), static_cast<rapidjson::SizeType>(key.size()), jdoc_->GetAllocator());
481
482 rapidjson::Value v;
483 if constexpr (std::is_same_v<T, std::string>)
484 {
485 v.SetString(val.c_str(), static_cast<rapidjson::SizeType>(val.size()), jdoc_->GetAllocator());
486 }
487 else if constexpr (std::is_same_v<T, char>)
488 {
489 char buf[2] = {val, '\0'};
490 v.SetString(buf, static_cast<rapidjson::SizeType>(sizeof(char)), jdoc_->GetAllocator());
491 }
492 else
493 {
494 v.Set(val);
495 }
496
497 jvalue_->AddMember(std::move(name), std::move(v), jdoc_->GetAllocator());
498
499 return *this;
500 }
