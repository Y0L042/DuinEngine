75typedef enum RENDERDOC_CaptureOption
76{
77 // Allow the application to enable vsync
78 //
79 // Default - enabled
80 //
81 // 1 - The application can enable or disable vsync at will
82 // 0 - vsync is force disabled
83 eRENDERDOC_Option_AllowVSync = 0,
84
85 // Allow the application to enable fullscreen
86 //
87 // Default - enabled
88 //
89 // 1 - The application can enable or disable fullscreen at will
90 // 0 - fullscreen is force disabled
91 eRENDERDOC_Option_AllowFullscreen = 1,
92
93 // Record API debugging events and messages
94 //
95 // Default - disabled
96 //
97 // 1 - Enable built-in API debugging features and records the results into
98 // the capture, which is matched up with events on replay
99 // 0 - no API debugging is forcibly enabled
100 eRENDERDOC_Option_APIValidation = 2,
101 eRENDERDOC_Option_DebugDeviceMode = 2, // deprecated name of this enum
102
103 // Capture CPU callstacks for API events
104 //
105 // Default - disabled
106 //
107 // 1 - Enables capturing of callstacks
108 // 0 - no callstacks are captured
109 eRENDERDOC_Option_CaptureCallstacks = 3,
110
111 // When capturing CPU callstacks, only capture them from actions.
112 // This option does nothing without the above option being enabled
113 //
114 // Default - disabled
115 //
116 // 1 - Only captures callstacks for actions.
117 // Ignored if CaptureCallstacks is disabled
118 // 0 - Callstacks, if enabled, are captured for every event.
119 eRENDERDOC_Option_CaptureCallstacksOnlyDraws = 4,
120 eRENDERDOC_Option_CaptureCallstacksOnlyActions = 4,
121
122 // Specify a delay in seconds to wait for a debugger to attach, after
123 // creating or injecting into a process, before continuing to allow it to run.
124 //
125 // 0 indicates no delay, and the process will run immediately after injection
126 //
127 // Default - 0 seconds
128 //
129 eRENDERDOC_Option_DelayForDebugger = 5,
130
131 // Verify buffer access. This includes checking the memory returned by a Map() call to
132 // detect any out-of-bounds modification, as well as initialising buffers with undefined contents
133 // to a marker value to catch use of uninitialised memory.
134 //
135 // NOTE: This option is only valid for OpenGL and D3D11. Explicit APIs such as D3D12 and Vulkan do
136 // not do the same kind of interception & checking and undefined contents are really undefined.
137 //
138 // Default - disabled
139 //
140 // 1 - Verify buffer access
141 // 0 - No verification is performed, and overwriting bounds may cause crashes or corruption in
142 // RenderDoc.
143 eRENDERDOC_Option_VerifyBufferAccess = 6,
144
145 // The old name for eRENDERDOC_Option_VerifyBufferAccess was eRENDERDOC_Option_VerifyMapWrites.
146 // This option now controls the filling of uninitialised buffers with 0xdddddddd which was
147 // previously always enabled
148 eRENDERDOC_Option_VerifyMapWrites = eRENDERDOC_Option_VerifyBufferAccess,
149
150 // Hooks any system API calls that create child processes, and injects
151 // RenderDoc into them recursively with the same options.
152 //
153 // Default - disabled
154 //
155 // 1 - Hooks into spawned child processes
156 // 0 - Child processes are not hooked by RenderDoc
157 eRENDERDOC_Option_HookIntoChildren = 7,
158
159 // By default RenderDoc only includes resources in the final capture necessary
160 // for that frame, this allows you to override that behaviour.
161 //
162 // Default - disabled
163 //
164 // 1 - all live resources at the time of capture are included in the capture
165 // and available for inspection
166 // 0 - only the resources referenced by the captured frame are included
167 eRENDERDOC_Option_RefAllResources = 8,
168
169 // **NOTE**: As of RenderDoc v1.1 this option has been deprecated. Setting or
170 // getting it will be ignored, to allow compatibility with older versions.
171 // In v1.1 the option acts as if it's always enabled.
172 //
173 // By default RenderDoc skips saving initial states for resources where the
174 // previous contents don't appear to be used, assuming that writes before
175 // reads indicate previous contents aren't used.
176 //
177 // Default - disabled
178 //
179 // 1 - initial contents at the start of each captured frame are saved, even if
180 // they are later overwritten or cleared before being used.
181 // 0 - unless a read is detected, initial contents will not be saved and will
182 // appear as black or empty data.
183 eRENDERDOC_Option_SaveAllInitials = 9,
184
185 // In APIs that allow for the recording of command lists to be replayed later,
186 // RenderDoc may choose to not capture command lists before a frame capture is
187 // triggered, to reduce overheads. This means any command lists recorded once
188 // and replayed many times will not be available and may cause a failure to
189 // capture.
190 //
191 // NOTE: This is only true for APIs where multithreading is difficult or
192 // discouraged. Newer APIs like Vulkan and D3D12 will ignore this option
193 // and always capture all command lists since the API is heavily oriented
194 // around it and the overheads have been reduced by API design.
195 //
196 // 1 - All command lists are captured from the start of the application
197 // 0 - Command lists are only captured if their recording begins during
198 // the period when a frame capture is in progress.
199 eRENDERDOC_Option_CaptureAllCmdLists = 10,
200
201 // Mute API debugging output when the API validation mode option is enabled
202 //
203 // Default - enabled
204 //
205 // 1 - Mute any API debug messages from being displayed or passed through
206 // 0 - API debugging is displayed as normal
207 eRENDERDOC_Option_DebugOutputMute = 11,
208
209 // Option to allow vendor extensions to be used even when they may be
210 // incompatible with RenderDoc and cause corrupted replays or crashes.
211 //
212 // Default - inactive
213 //
214 // No values are documented, this option should only be used when absolutely
215 // necessary as directed by a RenderDoc developer.
216 eRENDERDOC_Option_AllowUnsupportedVendorExtensions = 12,
217
218 // Define a soft memory limit which some APIs may aim to keep overhead under where
219 // possible. Anything above this limit will where possible be saved directly to disk during
220 // capture.
221 // This will cause increased disk space use (which may cause a capture to fail if disk space is
222 // exhausted) as well as slower capture times.
223 //
224 // Not all memory allocations may be deferred like this so it is not a guarantee of a memory
225 // limit.
226 //
227 // Units are in MBs, suggested values would range from 200MB to 1000MB.
228 //
229 // Default - 0 Megabytes
230 eRENDERDOC_Option_SoftMemoryLimit = 13,
231} RENDERDOC_CaptureOption;

514// When choosing either a device pointer or a window handle to capture, you can pass NULL.
515// Passing NULL specifies a 'wildcard' match against anything. This allows you to specify
516// any API rendering to a specific window, or a specific API instance rendering to any window,
517// or in the simplest case of one window and one API, you can just pass NULL for both.
518//
519// In either case, if there are two or more possible matching (device,window) pairs it
520// is undefined which one will be captured.
521//
522// Note: for headless rendering you can pass NULL for the window handle and either specify
523// a device pointer or leave it NULL as above.

527// If there is no matching thing to capture (e.g. no supported API has been initialised),

531// even on separate devices and/oror windows.

564// RenderDoc API versions
565//
566
567// RenderDoc uses semantic versioning (http://semver.org/).
568//
569// MAJOR version is incremented when incompatible API changes happen.
570// MINOR version is incremented when functionality is added in a backwards-compatible manner.
571// PATCH version is incremented when backwards-compatible bug fixes happen.
572//
573// Note that this means the API returned can be higher than the one you might have requested.
574// e.g. if you are running against a newer RenderDoc that supports 1.0.1, it will be returned
575// instead of 1.0.0. You can check this with the GetAPIVersion entry point
576typedef enum RENDERDOC_Version
577{
578 eRENDERDOC_API_Version_1_0_0 = 10000, // RENDERDOC_API_1_0_0 = 1 00 00
579 eRENDERDOC_API_Version_1_0_1 = 10001, // RENDERDOC_API_1_0_1 = 1 00 01
580 eRENDERDOC_API_Version_1_0_2 = 10002, // RENDERDOC_API_1_0_2 = 1 00 02
581 eRENDERDOC_API_Version_1_1_0 = 10100, // RENDERDOC_API_1_1_0 = 1 01 00
582 eRENDERDOC_API_Version_1_1_1 = 10101, // RENDERDOC_API_1_1_1 = 1 01 01
