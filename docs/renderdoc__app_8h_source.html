72 typedef enum RENDERDOC_CaptureOption
73 {
74 // Allow the application to enable vsync
75 //
76 // Default - enabled
77 //
78 // 1 - The application can enable or disable vsync at will
79 // 0 - vsync is force disabled
80 eRENDERDOC_Option_AllowVSync = 0,
81
82 // Allow the application to enable fullscreen
83 //
84 // Default - enabled
85 //
86 // 1 - The application can enable or disable fullscreen at will
87 // 0 - fullscreen is force disabled
88 eRENDERDOC_Option_AllowFullscreen = 1,
89
90 // Record API debugging events and messages
91 //
92 // Default - disabled
93 //
94 // 1 - Enable built-in API debugging features and records the results into
95 // the capture, which is matched up with events on replay
96 // 0 - no API debugging is forcibly enabled
97 eRENDERDOC_Option_APIValidation = 2,
98 eRENDERDOC_Option_DebugDeviceMode = 2, // deprecated name of this enum
99
100 // Capture CPU callstacks for API events
101 //
102 // Default - disabled
103 //
104 // 1 - Enables capturing of callstacks
105 // 0 - no callstacks are captured
106 eRENDERDOC_Option_CaptureCallstacks = 3,
107
108 // When capturing CPU callstacks, only capture them from actions.
109 // This option does nothing without the above option being enabled
110 //
111 // Default - disabled
112 //
113 // 1 - Only captures callstacks for actions.
114 // Ignored if CaptureCallstacks is disabled
115 // 0 - Callstacks, if enabled, are captured for every event.
116 eRENDERDOC_Option_CaptureCallstacksOnlyDraws = 4,
117 eRENDERDOC_Option_CaptureCallstacksOnlyActions = 4,
118
119 // Specify a delay in seconds to wait for a debugger to attach, after
120 // creating or injecting into a process, before continuing to allow it to run.
121 //
122 // 0 indicates no delay, and the process will run immediately after injection
123 //
124 // Default - 0 seconds
125 //
126 eRENDERDOC_Option_DelayForDebugger = 5,
127
128 // Verify buffer access. This includes checking the memory returned by a Map() call to
129 // detect any out-of-bounds modification, as well as initialising buffers with undefined contents
130 // to a marker value to catch use of uninitialised memory.
131 //
132 // NOTE: This option is only valid for OpenGL and D3D11. Explicit APIs such as D3D12 and Vulkan do
133 // not do the same kind of interception & checking and undefined contents are really undefined.
134 //
135 // Default - disabled
136 //
137 // 1 - Verify buffer access
138 // 0 - No verification is performed, and overwriting bounds may cause crashes or corruption in
139 // RenderDoc.
140 eRENDERDOC_Option_VerifyBufferAccess = 6,
141
142 // The old name for eRENDERDOC_Option_VerifyBufferAccess was eRENDERDOC_Option_VerifyMapWrites.
143 // This option now controls the filling of uninitialised buffers with 0xdddddddd which was
144 // previously always enabled
145 eRENDERDOC_Option_VerifyMapWrites = eRENDERDOC_Option_VerifyBufferAccess,
146
147 // Hooks any system API calls that create child processes, and injects
148 // RenderDoc into them recursively with the same options.
149 //
150 // Default - disabled
151 //
152 // 1 - Hooks into spawned child processes
153 // 0 - Child processes are not hooked by RenderDoc
154 eRENDERDOC_Option_HookIntoChildren = 7,
155
156 // By default RenderDoc only includes resources in the final capture necessary
157 // for that frame, this allows you to override that behaviour.
158 //
159 // Default - disabled
160 //
161 // 1 - all live resources at the time of capture are included in the capture
162 // and available for inspection
163 // 0 - only the resources referenced by the captured frame are included
164 eRENDERDOC_Option_RefAllResources = 8,
165
166 // **NOTE**: As of RenderDoc v1.1 this option has been deprecated. Setting or
167 // getting it will be ignored, to allow compatibility with older versions.
168 // In v1.1 the option acts as if it's always enabled.
169 //
170 // By default RenderDoc skips saving initial states for resources where the
171 // previous contents don't appear to be used, assuming that writes before
172 // reads indicate previous contents aren't used.
173 //
174 // Default - disabled
175 //
176 // 1 - initial contents at the start of each captured frame are saved, even if
177 // they are later overwritten or cleared before being used.
178 // 0 - unless a read is detected, initial contents will not be saved and will
179 // appear as black or empty data.
180 eRENDERDOC_Option_SaveAllInitials = 9,
181
182 // In APIs that allow for the recording of command lists to be replayed later,
183 // RenderDoc may choose to not capture command lists before a frame capture is
184 // triggered, to reduce overheads. This means any command lists recorded once
185 // and replayed many times will not be available and may cause a failure to
186 // capture.
187 //
188 // NOTE: This is only true for APIs where multithreading is difficult or
189 // discouraged. Newer APIs like Vulkan and D3D12 will ignore this option
190 // and always capture all command lists since the API is heavily oriented
191 // around it and the overheads have been reduced by API design.
192 //
193 // 1 - All command lists are captured from the start of the application
194 // 0 - Command lists are only captured if their recording begins during
195 // the period when a frame capture is in progress.
196 eRENDERDOC_Option_CaptureAllCmdLists = 10,
197
198 // Mute API debugging output when the API validation mode option is enabled
199 //
200 // Default - enabled
201 //
202 // 1 - Mute any API debug messages from being displayed or passed through
203 // 0 - API debugging is displayed as normal
204 eRENDERDOC_Option_DebugOutputMute = 11,
205
206 // Option to allow vendor extensions to be used even when they may be
207 // incompatible with RenderDoc and cause corrupted replays or crashes.
208 //
209 // Default - inactive
210 //
211 // No values are documented, this option should only be used when absolutely
212 // necessary as directed by a RenderDoc developer.
213 eRENDERDOC_Option_AllowUnsupportedVendorExtensions = 12,
214
215 // Define a soft memory limit which some APIs may aim to keep overhead under where
216 // possible. Anything above this limit will where possible be saved directly to disk during
217 // capture.
218 // This will cause increased disk space use (which may cause a capture to fail if disk space is
219 // exhausted) as well as slower capture times.
220 //
221 // Not all memory allocations may be deferred like this so it is not a guarantee of a memory
222 // limit.
223 //
224 // Units are in MBs, suggested values would range from 200MB to 1000MB.
225 //
226 // Default - 0 Megabytes
227 eRENDERDOC_Option_SoftMemoryLimit = 13,
228 } RENDERDOC_CaptureOption;

559 // RenderDoc API versions
560 //
561
562 // RenderDoc uses semantic versioning (http://semver.org/).
563 //
564 // MAJOR version is incremented when incompatible API changes happen.
565 // MINOR version is incremented when functionality is added in a backwards-compatible manner.
566 // PATCH version is incremented when backwards-compatible bug fixes happen.
567 //
568 // Note that this means the API returned can be higher than the one you might have requested.
569 // e.g. if you are running against a newer RenderDoc that supports 1.0.1, it will be returned
570 // instead of 1.0.0. You can check this with the GetAPIVersion entry point
571 typedef enum RENDERDOC_Version
572 {
573 eRENDERDOC_API_Version_1_0_0 = 10000, // RENDERDOC_API_1_0_0 = 1 00 00
574 eRENDERDOC_API_Version_1_0_1 = 10001, // RENDERDOC_API_1_0_1 = 1 00 01
575 eRENDERDOC_API_Version_1_0_2 = 10002, // RENDERDOC_API_1_0_2 = 1 00 02
576 eRENDERDOC_API_Version_1_1_0 = 10100, // RENDERDOC_API_1_1_0 = 1 01 00
577 eRENDERDOC_API_Version_1_1_1 = 10101, // RENDERDOC_API_1_1_1 = 1 01 01
578 eRENDERDOC_API_Version_1_1_2 = 10102, // RENDERDOC_API_1_1_2 = 1 01 02
