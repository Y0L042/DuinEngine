362 static void SetGlobalTransform(flecs::entity e, Transform3D tx)
363 {
364 if (!e.is_valid() || !e.has<Transform3D>())
365 {
366 DN_CORE_WARN("STATIC Entity not valid, or does not have Transform3D!");
367 return;
368 }
369 Transform3D *comp = e.try_get_mut<Transform3D>();
370 if (!comp)
371 {
372 return;
373 }
374
375 flecs::entity parent = e.parent();
376 if (parent.is_valid() && parent.has<Transform3D>())
377 {
378 Transform3D parentGlobal = GetGlobalTransform(parent);
379
380 // Compute local position.
381 // globalPos = parentGlobal.position + rotate(parentGlobal.rotation, localPos * parentGlobal.scale)
382 // Thus: localPos = inverse(parentGlobal.rotation) * (globalPos - parentGlobal.position) /
383 // parentGlobal.scale
384 Vector3 offset = Vector3Subtract(tx.GetPosition(), parentGlobal.GetPosition());
385 Quaternion invParentRot = QuaternionInvert(parentGlobal.GetRotation());
386 Vector3 localPosUnscaled = Vector3RotateByQuaternion(offset, invParentRot);
387 Vector3 localPos = Vector3Divide(localPosUnscaled, parentGlobal.GetScale());
388
389 // Compute local scale.
390 // Assuming scale is applied component-wise:
391 Vector3 localScale = Vector3Divide(tx.GetScale(), parentGlobal.GetScale());
392
393 // Compute local rotation.
394 // globalRot = parentGlobal.rotation * localRot, so:
395 Quaternion localRot = QuaternionMultiply(QuaternionInvert(parentGlobal.GetRotation()), tx.GetRotation());
396
397 comp->SetPosition(localPos);
398 comp->SetScale(localScale);
399 comp->SetRotation(localRot);
400 }
401 else
402 {
403 // No parent: local transform equals desired global transform.
404 comp->SetPosition(tx.GetPosition());
405 comp->SetScale(tx.GetScale());
406 comp->SetRotation(tx.GetRotation());
407 }
408 // Update caches immediately.
409 comp->UpdateGlobalPositionCache(tx.GetPosition());
410 comp->UpdateGlobalScaleCache(tx.GetScale());
411 comp->UpdateGlobalRotationCache(tx.GetRotation());
412 }

414 static Transform3D GetGlobalTransform(flecs::entity e)
415 {
416 if (!e.is_valid() || !e.has<Transform3D>())
417 {
418 return Transform3D();
419 }
420 Transform3D *tx = e.try_get_mut<Transform3D>();
421 if (!tx)
422 {
423 return Transform3D();
424 }
425
426 flecs::entity parent = e.parent();
427 if (parent.is_valid() && parent.has<Transform3D>())
428 {
429 // Get parent's global transform recursively.
430 Transform3D parentGlobal = GetGlobalTransform(parent);
431
432 // Compute global position:
433 // globalPos = parentGlobal.position + rotate(parentGlobal.rotation, localPos * parentGlobal.scale)
434 Vector3 scaledLocalPos = Vector3Multiply(tx->GetPosition(), parentGlobal.GetScale());
435 Vector3 rotatedPos = Vector3RotateByQuaternion(scaledLocalPos, parentGlobal.GetRotation());
436 Vector3 globalPos = Vector3Add(parentGlobal.GetPosition(), rotatedPos);
437
438 // Compute global scale.
439 Vector3 globalScale = Vector3Multiply(parentGlobal.GetScale(), tx->GetScale());
440
441 // Compute global rotation.
442 Quaternion globalRot = QuaternionMultiply(parentGlobal.GetRotation(), tx->GetRotation());
443
444 // Update the caches.
445 tx->UpdateGlobalPositionCache(globalPos);
446 tx->UpdateGlobalScaleCache(globalScale);
447 tx->UpdateGlobalRotationCache(globalRot);
448
449 return Transform3D(globalPos, globalRot, e);
450 }
451 else
452 {
453 // No parent: global transform equals local transform.
454 tx->UpdateGlobalPositionCache(tx->GetPosition());
455 tx->UpdateGlobalScaleCache(tx->GetScale());
456 tx->UpdateGlobalRotationCache(tx->GetRotation());
457
458 return *tx;
459 }
460 }

929 static Transform3D GetGlobalTransformRecursively(flecs::world &world, flecs::entity e)
930 {
931 if (!e.is_valid())
932 {
933 return Transform3D();
934 }
935 const Transform3D *tx = e.try_get<Transform3D>();
936 if (!tx)
937 {
938 return Transform3D();
939 }
940 if (!e.parent().is_valid())
941 {
942 return *tx;
943 }
944
945 Transform3D parentTx = GetGlobalTransformRecursively(world, e.parent());
946 Transform3D resultTx(Vector3Add(tx->GetPosition(), parentTx.GetPosition()),
947 Vector3Multiply(tx->GetScale(), parentTx.GetScale()),
948 QuaternionAdd(tx->GetRotation(), parentTx.GetRotation()));
