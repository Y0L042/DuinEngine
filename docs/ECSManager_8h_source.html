310 static void SetGlobalTransform(flecs::entity e, Transform3D tx)
311 {
312 if (!e.is_valid() || !e.has<Transform3D>())
313 {
314 DN_CORE_WARN("STATIC Entity not valid, or does not have Transform3D!");
315 return;
316 }
317 Transform3D *comp = e.try_get_mut<Transform3D>();
318 if (!comp)
319 {
320 return;
321 }
322
323 flecs::entity parent = e.parent();
324 if (parent.is_valid() && parent.has<Transform3D>())
325 {
326 Transform3D parentGlobal = GetGlobalTransform(parent);
327
328 // Compute local position.
329 // globalPos = parentGlobal.position + rotate(parentGlobal.rotation, localPos * parentGlobal.scale)
330 // Thus: localPos = inverse(parentGlobal.rotation) * (globalPos - parentGlobal.position) /
331 // parentGlobal.scale
332 Vector3 offset = Vector3Subtract(tx.GetPosition(), parentGlobal.GetPosition());
333 Quaternion invParentRot = QuaternionInvert(parentGlobal.GetRotation());
334 Vector3 localPosUnscaled = Vector3RotateByQuaternion(offset, invParentRot);
335 Vector3 localPos = Vector3Divide(localPosUnscaled, parentGlobal.GetScale());
336
337 // Compute local scale.
338 // Assuming scale is applied component-wise:
339 Vector3 localScale = Vector3Divide(tx.GetScale(), parentGlobal.GetScale());
340
341 // Compute local rotation.
342 // globalRot = parentGlobal.rotation * localRot, so:
343 Quaternion localRot = QuaternionMultiply(QuaternionInvert(parentGlobal.GetRotation()), tx.GetRotation());
344
345 comp->SetPosition(localPos);
346 comp->SetScale(localScale);
347 comp->SetRotation(localRot);
348 }
349 else
350 {
351 // No parent: local transform equals desired global transform.
352 comp->SetPosition(tx.GetPosition());
353 comp->SetScale(tx.GetScale());
354 comp->SetRotation(tx.GetRotation());
355 }
356 // Update caches immediately.
357 comp->UpdateGlobalPositionCache(tx.GetPosition());
358 comp->UpdateGlobalScaleCache(tx.GetScale());
359 comp->UpdateGlobalRotationCache(tx.GetRotation());
360 }

362 static Transform3D GetGlobalTransform(flecs::entity e)
363 {
364 if (!e.is_valid() || !e.has<Transform3D>())
365 {
366 return Transform3D();
367 }
368 Transform3D *tx = e.try_get_mut<Transform3D>();
369 if (!tx)
370 {
371 return Transform3D();
372 }
373
374 flecs::entity parent = e.parent();
375 if (parent.is_valid() && parent.has<Transform3D>())
376 {
377 // Get parent's global transform recursively.
378 Transform3D parentGlobal = GetGlobalTransform(parent);
379
380 // Compute global position:
381 // globalPos = parentGlobal.position + rotate(parentGlobal.rotation, localPos * parentGlobal.scale)
382 Vector3 scaledLocalPos = Vector3Multiply(tx->GetPosition(), parentGlobal.GetScale());
383 Vector3 rotatedPos = Vector3RotateByQuaternion(scaledLocalPos, parentGlobal.GetRotation());
384 Vector3 globalPos = Vector3Add(parentGlobal.GetPosition(), rotatedPos);
385
386 // Compute global scale.
387 Vector3 globalScale = Vector3Multiply(parentGlobal.GetScale(), tx->GetScale());
388
389 // Compute global rotation.
390 Quaternion globalRot = QuaternionMultiply(parentGlobal.GetRotation(), tx->GetRotation());
391
392 // Update the caches.
393 tx->UpdateGlobalPositionCache(globalPos);
394 tx->UpdateGlobalScaleCache(globalScale);
395 tx->UpdateGlobalRotationCache(globalRot);
396
397 return Transform3D(globalPos, globalRot, e);
398 }
399 else
400 {
401 // No parent: global transform equals local transform.
402 tx->UpdateGlobalPositionCache(tx->GetPosition());
403 tx->UpdateGlobalScaleCache(tx->GetScale());
404 tx->UpdateGlobalRotationCache(tx->GetRotation());
405
406 return *tx;
407 }
408 }

806 static Vector3 GetGlobalPositionRecursively(flecs::world &world, flecs::entity e)
807 {

863 static Transform3D GetGlobalTransformRecursively(flecs::world &world, flecs::entity e)
864 {
865 if (!e.is_valid())
866 {
867 return Transform3D();
868 }
869 const Transform3D *tx = e.try_get<Transform3D>();
870 if (!tx)
871 {
872 return Transform3D();
873 }
874 if (!e.parent().is_valid())
875 {
876 return *tx;
877 }
878
879 Transform3D parentTx = GetGlobalTransformRecursively(world, e.parent());
880 Transform3D resultTx(Vector3Add(tx->GetPosition(), parentTx.GetPosition()),
881 Vector3Multiply(tx->GetScale(), parentTx.GetScale()),
882 QuaternionAdd(tx->GetRotation(), parentTx.GetRotation()));
883
884 return resultTx;
