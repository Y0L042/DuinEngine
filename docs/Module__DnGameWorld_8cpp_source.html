12// Creates a ScriptGameWorld backed by the daslang class instance.
13// Returns a void* handle stored in ScriptMemory.
14static void *dn_create_gameworld_impl(void *classPtr, const das::StructInfo *info, das::Context *context)
15{
16 auto obj = std::make_shared<ScriptGameWorld>((char *)classPtr, info, context);
17 DN_CORE_INFO("dn_create_gameworld_impl: created ScriptGameWorld for '{}'", info->name);
18
19 auto *dnCtx = static_cast<duin::ScriptContext *>(context);
20 void *handle = static_cast<void *>(dnCtx->scriptMemory->Add(obj));
21
22 // Register as the active game world in the context so other systems can find it.
23 dnCtx->gameWorld = obj.get();
24
25 return handle;
26}

35static void dn_destroy_gameworld_impl(void *handle, das::Context *context)
36{
37 if (!handle)
38 return;
39 DN_CORE_INFO("dn_destroy_gameworld_impl: destroying ScriptGameWorld");
40 auto *dnCtx = static_cast<duin::ScriptContext *>(context);
41 if (dnCtx->gameWorld == static_cast<duin::GameWorld *>(handle))
42 dnCtx->gameWorld = nullptr;
43 dnCtx->scriptMemory->Remove(handle);
44}
