25void DebugConsole::Log(const char* format, ...) {
26 std::lock_guard<std::mutex> lock(mutex_);
27
28 // Buffer to store the formatted string
29 char buffer[1024]; // Adjust size as needed
30
31 // Initialize variadic arguments
32 va_list args;
33 va_start(args, format);
34
35 // Format the message using vsnprintf
36 vsnprintf(buffer, sizeof(buffer), format, args);
37
38 // End variadic arguments
39 va_end(args);
40
41 // Get time since game start
42 std::chrono::steady_clock::time_point currentTime = std::chrono::steady_clock::now();
43 std::chrono::duration<double> elapsed = currentTime - startTime; // Floating-point seconds
44
45 // Format the log message with the timestamp to 1 decimal place
46 std::ostringstream timestampedMessage;
47 timestampedMessage << "[" << std::fixed << std::setprecision(1) << elapsed.count() << "s] " << buffer;
48
49 logMessages_.emplace_back(timestampedMessage.str());
50 newMessageAdded = 1;
51}

132 }
133
134
135 // Scroll to bottom if new messages are added

137 ImGui::SetScrollHereY(1.0f);
138 newMessageAdded = false;
139 }
140 ImGui::EndChild();

142 ImGui::End();

155void DebugConsole::LogEx(LogLevel level, const char* format, ...) {
156 char buffer[1024];
157 va_list args;
158 va_start(args, format);
159 vsnprintf(buffer, sizeof(buffer), format, args);
160 va_end(args);
161
162
163 // Log to the main console logger
164 switch (level) {
165 case LogLevel::Fatal:
166 DN_FATAL("{0}", buffer);
167 break;
168 case LogLevel::Error:
169 DN_ERROR("{0}", buffer);
170 break;
171 case LogLevel::Warn:
172 DN_WARN("{0}", buffer);
173 break;
174 case LogLevel::Info:
175 DN_INFO("{0}", buffer);
176 break;
177 case LogLevel::Trace:
178 DN_TRACE("{0}", buffer);
179 break;
180 }
181 this->Log("%s", buffer);
182}

201void DebugWatchlist::Post(const std::string description, const std::string format, ...)
202{
203 std::lock_guard<std::mutex> lock(mutex_);
204
205 // Initialize variadic arguments
206 va_list args;
207 va_start(args, format);
208
209 // Estimate required buffer size
210 // +1 for null terminator
211 int size = std::vsnprintf(nullptr, 0, format.c_str(), args) + 1;
212 va_end(args); // End args before restarting
213
214 std::vector<char> buffer(size);
215 // Restart args for actual formatting
216 va_start(args, format);
217 std::vsnprintf(buffer.data(), buffer.size(), format.c_str(), args);
218 va_end(args);
219 // End variadic arguments
220
221 // Convert buffer to std::string
222 std::string value(buffer.data());
223
224 if (watchlist.find(description) == watchlist.end()) {
225 watchlist[description] = value;
226 order.push_back(description);
227 } else {
228 watchlist[description] = value;
229 }
