30PhysicsDebugRendererCore::Batch PhysicsDebugRendererCore::CreateTriangleBatch(
31 const Vertex *inVertices, int inVertexCount, const uint32_t *inIndices, int inIndexCount)
32{
33 BatchImpl *batch = new BatchImpl;
34 if (!inVertices || inVertexCount == 0 || !inIndices || inIndexCount == 0)
35 return batch;
36
37 batch->mTriangles.resize(inIndexCount / 3);
38 for (int t = 0; t < inIndexCount / 3; ++t)
39 {
40 Triangle &tri = batch->mTriangles[t];
41 tri.mV[0] = inVertices[inIndices[t * 3 + 0]];
42 tri.mV[1] = inVertices[inIndices[t * 3 + 1]];
43 tri.mV[2] = inVertices[inIndices[t * 3 + 2]];
44 }
