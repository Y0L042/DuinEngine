153std::string duin::fs::MapVirtualToSystemPath(const std::string &path)
154{
155 // Virtual paths must be at least "xxx://" (6 chars minimum)
156 // Paths received from SDL contains trailing '/'
157 if (path.size() < vpathSize - 1)
158 return INVALID_PATH;
159
160 std::string drive = path.substr(0, vpathSize);
161 std::string sysPath = INVALID_PATH;
162
163 if (drive == APPROOT)
164 {
165 sysPath = GetBasePath() + path.substr(vpathSize, path.size());
166 }
167
168 if (drive == APPDATA)
169 {
170 // Only map if ORG and APP have been set to non-empty values
171 if (ORG.empty() || APP.empty())
172 {
173 sysPath = INVALID_PATH;
174 }
175 else
176 {
177 sysPath = GetPrefPath(ORG, APP) + path.substr(vpathSize, path.size());
178 }
179 }
180
181 if (drive == USRDATA)
182 {
183 sysPath = INVALID_PATH;
184 }
185
