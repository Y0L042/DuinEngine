175bool duin::Application::ProcessFrame(double &deltaTime, double &physicsCurrentTime, double &physicsPreviousTime,
176 double &physicsAccumTime)
177{
178#ifdef DN_DEBUG
179 if (!debugIsGamePaused_)
180 {
181#endif /* DN_DEBUG */
182
183 /* Do not run game when minimized */
184 if (PAUSE_ON_MINIMIZE && SDL_GetWindowFlags(sdlWindow) && SDL_WINDOW_MINIMIZED)
185 {
186 duin::DelayProcessMilli(100);
187 return !gameShouldQuit;
188 }
189
190 double frameStartTime = duin::GetTicks();
191
192 EnginePreFrame();
193
194 /* If custom ImGui path is set, manually save to memory/disk */
195 if (USE_CUSTOM_IMGUI_INI_PATH && ImGui::GetIO().WantSaveIniSettings)
196 {
197 ImGui::SaveIniSettingsToDisk(IMGUI_INI_PATH.c_str());
198 }
199
200 ProcessEvents();
201
202 RunUpdate(deltaTime);
203
204 RunPhysics(physicsCurrentTime, physicsPreviousTime, physicsAccumTime);
205
206 /* Skip rendering when minimized */
207 SDL_Event e;
208 SDL_zero(e);
209 if (SDL_GetWindowFlags(sdlWindow) && e.type == SDL_WINDOW_MINIMIZED)
210 {
211 ImGui::Render();
212 duin::DelayProcessMilli(100);
213 return !gameShouldQuit;
214 }
215
216 /* Skip rendering if window size is 0 */
217 int displayWidth, displayHeight;
218 ::SDL_GetWindowSize(sdlWindow, &displayWidth, &displayHeight);
219 if (!(displayWidth && displayHeight))
220 {
221 duin::DelayProcessMilli(100);
222 return !gameShouldQuit;
223 }
224
225 RunRender();
226
227 EnginePostFrame();
228
229 double frameEndTime = duin::GetTicks();
230 double deltaDrawTime = frameEndTime - frameStartTime;
231
232 if (TARGET_RENDER_FRAMERATE > 0)
233 {
234 double targetFrameTime = 1.0 / (double)TARGET_RENDER_FRAMERATE;
235 double waitTime = targetFrameTime - deltaDrawTime;
236
237 if (waitTime > 0.0)
238 {
239 duin::DelayProcess((float)waitTime);
240 }
241 }
242
243 deltaTime = duin::GetTicks() - frameStartTime;
244 renderFrameTime = deltaTime;
245
246#ifdef DN_DEBUG
247 }
248 EngineDebug();
249 Debug();
250 EnginePostDebug();
251#endif /* DN_DEBUG */
252
253 return !gameShouldQuit;
254}
