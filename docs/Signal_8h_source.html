118 std::function<void(types...)> callback)

120 UUID id = Connect(std::move(callback));
121 std::weak_ptr<SignalImpl> weakSelf = self;
122 auto conn = std::shared_ptr<ScopedConnection>(new ScopedConnection(id, [weakSelf, id]() {
123 if (auto impl = weakSelf.lock())

130 void Disconnect(UUID uuid)
131 {
132 for (auto &[id, weakConn] : scopedConnections_)
133 {
134 if (id == uuid)
135 {
136 if (auto conn = weakConn.lock())
137 conn->Invalidate();
138 break;
139 }
140 }
141 std::erase_if(scopedConnections_, [&](const auto &p) { return p.first == uuid; });
142
143 if (emitExecuting_)

147 }

163 for (auto &[id, weakConn] : scopedConnections_)

176 bool Emit(types... args)
177 {
178 DN_CORE_WARN_IF(emitExecuting_,
179 "Recursive infinite emition detected! Do not re-emit signal while it is still emitting");
180 //DN_CORE_ASSERT(!emitExecuting_,
181 // "Recursive infinite emition detected! Do not re-emit signal while it is still emitting");
182
183 if (!emitExecuting_)
184 {
185 recursionLimitHit = false;
186 }
187
188 ++recursiveEmissionCount;
189 if (recursiveEmissionCount >= 30)
190 {
191 DN_CORE_FATAL("Recursive emittion exceeded 30! Preventing recursion loop.");
192 --recursiveEmissionCount;
193 recursionLimitHit = true;
194 return false;
195 }
196
197 emitExecuting_ = true;
198
199 for (size_t i = 0; i < callbacks_.size(); i++)
200 {
201 auto &[id, fn] = callbacks_[i];
202 if (fn)
203 fn(args...);
204 else
205 Disconnect(id);
206 }
207
208 emitExecuting_ = false;
209
210 if (queueEraseAll_)
211 {
212 callbacks_.clear();
213 queueEraseAll_ = false;
214 }
215 else
216 {
217 EraseQueue();
218 }
219
220 bool res = recursionLimitHit;
221
222 --recursiveEmissionCount;
223 return !res;
224 }
225

229 }

246 std::erase_if(callbacks_, [uuid](const Listener &l) { return l.first == uuid; });

248 eraseQueue_.clear();
