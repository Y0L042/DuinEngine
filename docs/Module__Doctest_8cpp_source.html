84// ─── Test discovery ──────────────────────────────────────────────────────────
85// daslang scripts call dn_register_test(suite, name, fn_name, at) during their
86// module-level initialisation (or in game_ready / any early call). We store
87// the context + SimFunction* in a static entry and register a plain trampoline
88// with doctest so the test appears in normal test discovery and is run by the
89// doctest runner like any C++ TEST_CASE.
90
91struct DasTestEntry
92{
93 std::string suite;
94 std::string name;
95 std::string file;
96 int line = 0;
97 std::shared_ptr<das::Context> context;
98 das::SimFunction *fn = nullptr;
99};
100
101static std::vector<DasTestEntry> g_dasTests;
102
103// Each registered test needs a unique static void(*)() trampoline. We
104// generate them on demand from a fixed-size pool; increase DAS_TEST_POOL_SIZE
105// if you need more than this many daslang tests per process.
106static constexpr int DAS_TEST_POOL_SIZE = 256;
107
108static int g_trampolineCount = 0;
109static int g_trampolineIndex[DAS_TEST_POOL_SIZE]; // maps trampoline slot → g_dasTests index
110
111template <int N>
112static void das_test_trampoline()
113{
114 const DasTestEntry &e = g_dasTests[g_trampolineIndex[N]];
115 e.context->evalWithCatch(e.fn, nullptr, nullptr);
116 if (auto ex = e.context->getException())
117 {
118 doctest::detail::MessageBuilder mb(e.file.c_str(), e.line,
119 doctest::assertType::DT_REQUIRE);
120 mb * ex;
121 if (mb.log())
122 mb.react();
123 }
124}
125
126// Build a compile-time table of trampoline function pointers.
127template <int... Ns>
128static doctest::detail::funcType trampoline_for(int n, std::integer_sequence<int, Ns...>)
129{
130 static doctest::detail::funcType table[] = {&das_test_trampoline<Ns>...};
131 return table[n];
132}
133
134static doctest::detail::funcType get_trampoline(int slot)
135{
136 return trampoline_for(slot, std::make_integer_sequence<int, DAS_TEST_POOL_SIZE>{});
137}
138
139// Called from daslang: dn_register_test(suite, name, fn_name, at)
140static void dn_register_test(const char *suite, const char *name, const char *fn_name,
141 das::LineInfoArg *at, das::Context *ctx)
142{
143 if (g_trampolineCount >= DAS_TEST_POOL_SIZE)
144 {
145 DN_CORE_ERROR("dn_register_test: pool exhausted (max {})", DAS_TEST_POOL_SIZE);
146 return;
147 }
148
149 das::SimFunction *fn = ctx->findFunction(fn_name ? fn_name : "");
150 if (!fn)
151 {
152 DN_CORE_ERROR("dn_register_test: function '{}' not found in script", fn_name ? fn_name : "");
153 return;
154 }
155
156 int slot = g_trampolineCount++;
157 g_trampolineIndex[slot] = (int)g_dasTests.size();
158
159 DasTestEntry entry;
160 entry.suite = suite ? suite : "";
161 entry.name = name ? name : "";
162 entry.file = location_file(at);
163 entry.line = location_line(at);
164 // Share ownership of the context so it stays alive while tests run.
165 entry.context = std::shared_ptr<das::Context>(ctx, [](das::Context *) {});
166 entry.fn = fn;
167 g_dasTests.push_back(std::move(entry));
168
169 doctest::detail::TestSuite ts;
170 ts.m_test_suite = g_dasTests.back().suite.c_str();
171
172 doctest::detail::TestCase tc(
173 get_trampoline(slot),
174 g_dasTests.back().file.c_str(),
175 (unsigned)g_dasTests.back().line,
176 ts);
177 tc * g_dasTests.back().name.c_str();
178
179 doctest::detail::regTest(tc);
180}
