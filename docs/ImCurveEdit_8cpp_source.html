66 static float smoothstep(float edge0, float edge1, float x)
67 {
68 x = ImClamp((x - edge0) / (edge1 - edge0), 0.0f, 1.0f);
69 return x * x * (3 - 2 * x);
70 }

72 static float distance(float x, float y, float x1, float y1, float x2, float y2)
73 {
74 float A = x - x1;
75 float B = y - y1;
76 float C = x2 - x1;
77 float D = y2 - y1;
78
79 float dot = A * C + B * D;
80 float len_sq = C * C + D * D;
81 float param = -1.f;
82 if (len_sq > FLT_EPSILON)
83 param = dot / len_sq;
84
85 float xx, yy;
86
87 if (param < 0.f) {
88 xx = x1;
89 yy = y1;
90 }
91 else if (param > 1.f) {
92 xx = x2;
93 yy = y2;
94 }
95 else {
96 xx = x1 + param * C;
97 yy = y1 + param * D;
98 }
99
100 float dx = x - xx;
101 float dy = y - yy;
102 return sqrtf(dx * dx + dy * dy);
103 }

105 static int DrawPoint(ImDrawList* draw_list, ImVec2 pos, const ImVec2 size, const ImVec2 offset, bool edited)
106 {
107 int ret = 0;
108 ImGuiIO& io = ImGui::GetIO();
109
110 static const ImVec2 localOffsets[4] = { ImVec2(1,0), ImVec2(0,1), ImVec2(-1,0), ImVec2(0,-1) };
111 ImVec2 offsets[4];
112 for (int i = 0; i < 4; i++)
113 {
114 offsets[i] = pos * size + localOffsets[i] * 4.5f + offset;
115 }
116
117 const ImVec2 center = pos * size + offset;
118 const ImRect anchor(center - ImVec2(5, 5), center + ImVec2(5, 5));
119 draw_list->AddConvexPolyFilled(offsets, 4, 0xFF000000);
120 if (anchor.Contains(io.MousePos))
121 {
122 ret = 1;
123 if (io.MouseDown[0])
124 ret = 2;
125 }
126 if (edited)
127 draw_list->AddPolyline(offsets, 4, 0xFFFFFFFF, true, 3.0f);
128 else if (ret)
129 draw_list->AddPolyline(offsets, 4, 0xFF80B0FF, true, 2.0f);
130 else
131 draw_list->AddPolyline(offsets, 4, 0xFF0080FF, true, 2.0f);
132
133 return ret;
134 }

136 int Edit(Delegate& delegate, const ImVec2& size, unsigned int id, const ImRect* clippingRect, ImVector<EditPoint>* selectedPoints)
137 {
138 static bool selectingQuad = false;
139 static ImVec2 quadSelection;
140 static int overCurve = -1;
141 static int movingCurve = -1;
142 static bool scrollingV = false;
143 static std::set<EditPoint> selection;
144 static bool overSelectedPoint = false;
145
146 int ret = 0;
147
148 ImGuiIO& io = ImGui::GetIO();
149 ImGui::PushStyleVar(ImGuiStyleVar_FramePadding, ImVec2(0, 0));
150 ImGui::PushStyleColor(ImGuiCol_Border, 0);
151 ImGui::BeginChild(id, size, ImGuiChildFlags_FrameStyle);
152 delegate.focused = ImGui::IsWindowFocused();
153 ImDrawList* draw_list = ImGui::GetWindowDrawList();
154 if (clippingRect)
155 draw_list->PushClipRect(clippingRect->Min, clippingRect->Max, true);
156
157 const ImVec2 offset = ImGui::GetCursorScreenPos() + ImVec2(0.f, size.y);
158 const ImVec2 ssize(size.x, -size.y);
159 const ImRect container(offset + ImVec2(0.f, ssize.y), offset + ImVec2(ssize.x, 0.f));
160 ImVec2& min = delegate.GetMin();
161 ImVec2& max = delegate.GetMax();
162
163 // handle zoom and VScroll
164 if (container.Contains(io.MousePos))
165 {
166 if (fabsf(io.MouseWheel) > FLT_EPSILON)
167 {
168 const float r = (io.MousePos.y - offset.y) / ssize.y;
169 float ratioY = ImLerp(min.y, max.y, r);
170 auto scaleValue = [&](float v) {
171 v -= ratioY;
172 v *= (1.f - io.MouseWheel * 0.05f);
173 v += ratioY;
174 return v;
175 };
176 min.y = scaleValue(min.y);
177 max.y = scaleValue(max.y);
178 }
179 if (!scrollingV && ImGui::IsMouseDown(2))
180 {
181 scrollingV = true;
182 }
183 }
184 ImVec2 range = max - min + ImVec2(1.f, 0.f); // +1 because of inclusive last frame
185
186 const ImVec2 viewSize(size.x, -size.y);
187 const ImVec2 sizeOfPixel = ImVec2(1.f, 1.f) / viewSize;
188 const size_t curveCount = delegate.GetCurveCount();
189
190 if (scrollingV)
191 {
192 float deltaH = io.MouseDelta.y * range.y * sizeOfPixel.y;
193 min.y -= deltaH;
194 max.y -= deltaH;
195 if (!ImGui::IsMouseDown(2))
196 scrollingV = false;
197 }
198
199 draw_list->AddRectFilled(offset, offset + ssize, delegate.GetBackgroundColor());
200
201 auto pointToRange = [&](ImVec2 pt) { return (pt - min) / range; };
202 auto rangeToPoint = [&](ImVec2 pt) { return (pt * range) + min; };
203
204 draw_list->AddLine(ImVec2(-1.f, -min.y / range.y) * viewSize + offset, ImVec2(1.f, -min.y / range.y) * viewSize + offset, 0xFF000000, 1.5f);
205 bool overCurveOrPoint = false;
206
207 int localOverCurve = -1;
208 // make sure highlighted curve is rendered last
209 int* curvesIndex = (int*)_malloca(sizeof(int) * curveCount);
210 for (size_t c = 0; c < curveCount; c++)
211 curvesIndex[c] = int(c);
212 int highLightedCurveIndex = -1;
213 if (overCurve != -1 && curveCount)
214 {
215 ImSwap(curvesIndex[overCurve], curvesIndex[curveCount - 1]);
216 highLightedCurveIndex = overCurve;
217 }
218
219 for (size_t cur = 0; cur < curveCount; cur++)
220 {
221 int c = curvesIndex[cur];
222 if (!delegate.IsVisible(c))
223 continue;
224 const size_t ptCount = delegate.GetPointCount(c);
225 if (ptCount < 1)
226 continue;
227 CurveType curveType = delegate.GetCurveType(c);
228 if (curveType == CurveNone)
229 continue;
230 const ImVec2* pts = delegate.GetPoints(c);
231 uint32_t curveColor = delegate.GetCurveColor(c);
232 if ((c == highLightedCurveIndex && selection.empty() && !selectingQuad) || movingCurve == c)
233 curveColor = 0xFFFFFFFF;
234
235 for (size_t p = 0; p < ptCount - 1; p++)
236 {
237 const ImVec2 p1 = pointToRange(pts[p]);
238 const ImVec2 p2 = pointToRange(pts[p + 1]);
239
240 if (curveType == CurveSmooth || curveType == CurveLinear)
241 {
242 size_t subStepCount = (curveType == CurveSmooth) ? 20 : 2;
243 float step = 1.f / float(subStepCount - 1);
244 for (size_t substep = 0; substep < subStepCount - 1; substep++)
245 {
246 float t = float(substep) * step;
247
248 const ImVec2 sp1 = ImLerp(p1, p2, t);
249 const ImVec2 sp2 = ImLerp(p1, p2, t + step);
250
251 const float rt1 = smoothstep(p1.x, p2.x, sp1.x);
252 const float rt2 = smoothstep(p1.x, p2.x, sp2.x);
253
254 const ImVec2 pos1 = ImVec2(sp1.x, ImLerp(p1.y, p2.y, rt1)) * viewSize + offset;
255 const ImVec2 pos2 = ImVec2(sp2.x, ImLerp(p1.y, p2.y, rt2)) * viewSize + offset;
256
257 if (distance(io.MousePos.x, io.MousePos.y, pos1.x, pos1.y, pos2.x, pos2.y) < 8.f && !scrollingV)
258 {
259 localOverCurve = int(c);
260 overCurve = int(c);
261 overCurveOrPoint = true;
262 }
263
264 draw_list->AddLine(pos1, pos2, curveColor, 1.3f);
265 } // substep
266 }
267 else if (curveType == CurveDiscrete)
268 {
269 ImVec2 dp1 = p1 * viewSize + offset;
270 ImVec2 dp2 = ImVec2(p2.x, p1.y) * viewSize + offset;
271 ImVec2 dp3 = p2 * viewSize + offset;
272 draw_list->AddLine(dp1, dp2, curveColor, 1.3f);
273 draw_list->AddLine(dp2, dp3, curveColor, 1.3f);
274
275 if ((distance(io.MousePos.x, io.MousePos.y, dp1.x, dp1.y, dp3.x, dp1.y) < 8.f ||
276 distance(io.MousePos.x, io.MousePos.y, dp3.x, dp1.y, dp3.x, dp3.y) < 8.f)
277 /*&& localOverCurve == -1*/)
278 {
279 localOverCurve = int(c);
280 overCurve = int(c);
281 overCurveOrPoint = true;
282 }
283 }
284 } // point loop
285
286 for (size_t p = 0; p < ptCount; p++)
287 {
288 const int drawState = DrawPoint(draw_list, pointToRange(pts[p]), viewSize, offset, (selection.find({ int(c), int(p) }) != selection.end() && movingCurve == -1 && !scrollingV));
289 if (drawState && movingCurve == -1 && !selectingQuad)
290 {
291 overCurveOrPoint = true;
292 overSelectedPoint = true;
293 overCurve = -1;
294 if (drawState == 2)
295 {
296 if (!io.KeyShift && selection.find({ int(c), int(p) }) == selection.end())
297 selection.clear();
298 selection.insert({ int(c), int(p) });
299 }
300 }
301 }
302 } // curves loop
303
304 if (localOverCurve == -1)
305 overCurve = -1;
306
307 // move selection
308 static bool pointsMoved = false;
309 static ImVec2 mousePosOrigin;
310 static std::vector<ImVec2> originalPoints;
311 if (overSelectedPoint && io.MouseDown[0])
312 {
313 if ((fabsf(io.MouseDelta.x) > 0.f || fabsf(io.MouseDelta.y) > 0.f) && !selection.empty())
314 {
315 if (!pointsMoved)
316 {
317 delegate.BeginEdit(0);
318 mousePosOrigin = io.MousePos;
319 originalPoints.resize(selection.size());
320 int index = 0;
321 for (auto& sel : selection)
322 {
323 const ImVec2* pts = delegate.GetPoints(sel.curveIndex);
324 originalPoints[index++] = pts[sel.pointIndex];
325 }
326 }
327 pointsMoved = true;
328 ret = 1;
329 auto prevSelection = selection;
330 int originalIndex = 0;
331 for (auto& sel : prevSelection)
332 {
333 const ImVec2 p = rangeToPoint(pointToRange(originalPoints[originalIndex]) + (io.MousePos - mousePosOrigin) * sizeOfPixel);
334 const int newIndex = delegate.EditPoint(sel.curveIndex, sel.pointIndex, p);
335 if (newIndex != sel.pointIndex)
336 {
337 selection.erase(sel);
338 selection.insert({ sel.curveIndex, newIndex });
339 }
340 originalIndex++;
341 }
342 }
343 }
344
345 if (overSelectedPoint && !io.MouseDown[0])
346 {
347 overSelectedPoint = false;
348 if (pointsMoved)
349 {
350 pointsMoved = false;
351 delegate.EndEdit();
352 }
353 }
354
355 // add point
356 if (overCurve != -1 && io.MouseDoubleClicked[0])
357 {
358 const ImVec2 np = rangeToPoint((io.MousePos - offset) / viewSize);
359 delegate.BeginEdit(overCurve);
360 delegate.AddPoint(overCurve, np);
361 delegate.EndEdit();
362 ret = 1;
363 }
364
365 // move curve
366
367 if (movingCurve != -1)
368 {
369 const size_t ptCount = delegate.GetPointCount(movingCurve);
370 const ImVec2* pts = delegate.GetPoints(movingCurve);
371 if (!pointsMoved)
372 {
373 mousePosOrigin = io.MousePos;
374 pointsMoved = true;
375 originalPoints.resize(ptCount);
376 for (size_t index = 0; index < ptCount; index++)
377 {
378 originalPoints[index] = pts[index];
379 }
380 }
381 if (ptCount >= 1)
382 {
383 for (size_t p = 0; p < ptCount; p++)
384 {
385 delegate.EditPoint(movingCurve, int(p), rangeToPoint(pointToRange(originalPoints[p]) + (io.MousePos - mousePosOrigin) * sizeOfPixel));
386 }
387 ret = 1;
388 }
389 if (!io.MouseDown[0])
390 {
391 movingCurve = -1;
392 pointsMoved = false;
393 delegate.EndEdit();
394 }
395 }
396 if (movingCurve == -1 && overCurve != -1 && ImGui::IsMouseClicked(0) && selection.empty() && !selectingQuad)
397 {
398 movingCurve = overCurve;
399 delegate.BeginEdit(overCurve);
400 }
401
402 // quad selection
403 if (selectingQuad)
404 {
405 const ImVec2 bmin = ImMin(quadSelection, io.MousePos);
406 const ImVec2 bmax = ImMax(quadSelection, io.MousePos);
407 draw_list->AddRectFilled(bmin, bmax, 0x40FF0000, 1.f);
408 draw_list->AddRect(bmin, bmax, 0xFFFF0000, 1.f);
409 const ImRect selectionQuad(bmin, bmax);
410 if (!io.MouseDown[0])
411 {
412 if (!io.KeyShift)
413 selection.clear();
414 // select everythnig is quad
415 for (size_t c = 0; c < curveCount; c++)
416 {
417 if (!delegate.IsVisible(c))
418 continue;
419
420 const size_t ptCount = delegate.GetPointCount(c);
421 if (ptCount < 1)
422 continue;
423
424 const ImVec2* pts = delegate.GetPoints(c);
425 for (size_t p = 0; p < ptCount; p++)
426 {
427 const ImVec2 center = pointToRange(pts[p]) * viewSize + offset;
428 if (selectionQuad.Contains(center))
429 selection.insert({ int(c), int(p) });
430 }
431 }
432 // done
433 selectingQuad = false;
434 }
435 }
436 if (!overCurveOrPoint && ImGui::IsMouseClicked(0) && !selectingQuad && movingCurve == -1 && !overSelectedPoint && container.Contains(io.MousePos))
437 {
438 selectingQuad = true;
439 quadSelection = io.MousePos;
440 }
441 if (clippingRect)
442 draw_list->PopClipRect();
443
444 ImGui::EndChild();
445 ImGui::PopStyleVar();
446 ImGui::PopStyleColor(1);
447
448 if (selectedPoints)
449 {
450 selectedPoints->resize(int(selection.size()));
451 int index = 0;
452 for (auto& point : selection)
453 (*selectedPoints)[index++] = point;
454 }
455 _freea(curvesIndex);
456 return ret;
457 }
