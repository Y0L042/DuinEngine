124 static std::string MakeQueryKey(const char *tag)

126 std::string key = tag;

132 template <typename... Comps, typename BuilderFn>
133 Query<Comps...> &GetOrBuildQuery(const char *tag, BuilderFn &&builderFn)
134 {
135 std::string key = MakeQueryKey<Comps...>(tag);
136 auto it = queryCache_.find(key);
137 if (it != queryCache_.end())
138 {
139 auto *ptr = std::any_cast<Query<Comps...>>(&it->second);
140 DN_CORE_ASSERT(ptr, "Query cache type mismatch");
141 return *ptr;
142 }
143 Query<Comps...> q = std::forward<BuilderFn>(builderFn)(*this);
144 auto [ins, ok] = queryCache_.emplace(key, std::move(q));
145 return *std::any_cast<Query<Comps...>>(&ins->second);
146 }
