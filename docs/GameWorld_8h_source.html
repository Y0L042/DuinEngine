121 template <typename... Comps>
122 static std::string MakeQueryKey(const char *tag)
123 {
124 std::string key = tag;
125 key += ":";

130 template <typename... Comps, typename BuilderFn>
131 Query<Comps...> &GetOrBuildQuery(const char *tag, BuilderFn &&builderFn)
132 {
133 std::string key = MakeQueryKey<Comps...>(tag);
134 auto it = queryCache_.find(key);
135 if (it != queryCache_.end())
136 {
137 auto *ptr = std::any_cast<Query<Comps...>>(&it->second);
138 DN_CORE_ASSERT(ptr, "Query cache type mismatch");
139 return *ptr;
140 }
141 Query<Comps...> q = std::forward<BuilderFn>(builderFn)(*this);
142 auto [ins, ok] = queryCache_.emplace(key, std::move(q));
143 return *std::any_cast<Query<Comps...>>(&ins->second);
144 }
