160static void ImGui_ImplSDL3_PlatformSetImeData(ImGuiContext*, ImGuiViewport* viewport, ImGuiPlatformImeData* data)
161{
162 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
163 SDL_WindowID window_id = (SDL_WindowID)(intptr_t)viewport->PlatformHandle;
164 SDL_Window* window = SDL_GetWindowFromID(window_id);
165 if ((data->WantVisible == false || bd->ImeWindow != window) && bd->ImeWindow != nullptr)
166 {
167 SDL_StopTextInput(bd->ImeWindow);
168 bd->ImeWindow = nullptr;
169 }
170 if (data->WantVisible)
171 {
172 SDL_Rect r;
173 r.x = (int)(data->InputPos.x - viewport->Pos.x);
174 r.y = (int)(data->InputPos.y - viewport->Pos.y + data->InputLineHeight);
175 r.w = 1;
176 r.h = (int)data->InputLineHeight;
177 SDL_SetTextInputArea(window, &r, 0);
178 SDL_StartTextInput(window);
179 bd->ImeWindow = window;
180 }
181}

350// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
351// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
352// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
353// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
354bool ImGui_ImplSDL3_ProcessEvent(const SDL_Event* event)
355{
356 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
357 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplSDL3_Init()?");
358 ImGuiIO& io = ImGui::GetIO();
359
360 switch (event->type)
361 {
362 case SDL_EVENT_MOUSE_MOTION:
363 {
364 if (ImGui_ImplSDL3_GetViewportForWindowID(event->motion.windowID) == nullptr)
365 return false;
366 ImVec2 mouse_pos((float)event->motion.x, (float)event->motion.y);
367 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
368 {
369 int window_x, window_y;
370 SDL_GetWindowPosition(SDL_GetWindowFromID(event->motion.windowID), &window_x, &window_y);
371 mouse_pos.x += window_x;
372 mouse_pos.y += window_y;
373 }
374 io.AddMouseSourceEvent(event->motion.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
375 io.AddMousePosEvent(mouse_pos.x, mouse_pos.y);
376 return true;
377 }
378 case SDL_EVENT_MOUSE_WHEEL:
379 {
380 if (ImGui_ImplSDL3_GetViewportForWindowID(event->wheel.windowID) == nullptr)
381 return false;
382 //IMGUI_DEBUG_LOG("wheel %.2f %.2f, precise %.2f %.2f\n", (float)event->wheel.x, (float)event->wheel.y, event->wheel.preciseX, event->wheel.preciseY);
383 float wheel_x = -event->wheel.x;
384 float wheel_y = event->wheel.y;
385 io.AddMouseSourceEvent(event->wheel.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
386 io.AddMouseWheelEvent(wheel_x, wheel_y);
387 return true;
388 }
389 case SDL_EVENT_MOUSE_BUTTON_DOWN:
390 case SDL_EVENT_MOUSE_BUTTON_UP:
391 {
392 if (ImGui_ImplSDL3_GetViewportForWindowID(event->button.windowID) == nullptr)
393 return false;
394 int mouse_button = -1;
395 if (event->button.button == SDL_BUTTON_LEFT) { mouse_button = 0; }
396 if (event->button.button == SDL_BUTTON_RIGHT) { mouse_button = 1; }
397 if (event->button.button == SDL_BUTTON_MIDDLE) { mouse_button = 2; }
398 if (event->button.button == SDL_BUTTON_X1) { mouse_button = 3; }
399 if (event->button.button == SDL_BUTTON_X2) { mouse_button = 4; }
400 if (mouse_button == -1)
401 break;
402 io.AddMouseSourceEvent(event->button.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
403 io.AddMouseButtonEvent(mouse_button, (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN));
404 bd->MouseButtonsDown = (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN) ? (bd->MouseButtonsDown | (1 << mouse_button)) : (bd->MouseButtonsDown & ~(1 << mouse_button));
405 return true;
406 }
407 case SDL_EVENT_TEXT_INPUT:
408 {
409 if (ImGui_ImplSDL3_GetViewportForWindowID(event->text.windowID) == nullptr)
410 return false;
411 io.AddInputCharactersUTF8(event->text.text);
412 return true;
413 }
414 case SDL_EVENT_KEY_DOWN:
415 case SDL_EVENT_KEY_UP:
416 {
417 if (ImGui_ImplSDL3_GetViewportForWindowID(event->key.windowID) == nullptr)
418 return false;
419 ImGui_ImplSDL3_UpdateKeyModifiers((SDL_Keymod)event->key.mod);
420 //IMGUI_DEBUG_LOG("SDL_EVENT_KEY_%s : key=%d ('%s'), scancode=%d ('%s'), mod=%X\n",
421 // (event->type == SDL_EVENT_KEY_DOWN) ? "DOWN" : "UP ", event->key.key, SDL_GetKeyName(event->key.key), event->key.scancode, SDL_GetScancodeName(event->key.scancode), event->key.mod);
422 ImGuiKey key = ImGui_ImplSDL3_KeyEventToImGuiKey(event->key.key, event->key.scancode);
423 io.AddKeyEvent(key, (event->type == SDL_EVENT_KEY_DOWN));
424 io.SetKeyEventNativeData(key, event->key.key, event->key.scancode, event->key.scancode); // To support legacy indexing (<1.87 user code). Legacy backend uses SDLK_*** as indices to IsKeyXXX() functions.
425 return true;
426 }
427 case SDL_EVENT_WINDOW_MOUSE_ENTER:
428 {
429 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
430 return false;
431 bd->MouseWindowID = event->window.windowID;
432 bd->MousePendingLeaveFrame = 0;
433 return true;
434 }
435 // - In some cases, when detaching a window from main viewport SDL may send SDL_WINDOWEVENT_ENTER one frame too late,
436 // causing SDL_WINDOWEVENT_LEAVE on previous frame to interrupt drag operation by clear mouse position. This is why
437 // we delay process the SDL_WINDOWEVENT_LEAVE events by one frame. See issue #5012 for details.
438 // FIXME: Unconfirmed whether this is still needed with SDL3.
439 case SDL_EVENT_WINDOW_MOUSE_LEAVE:
440 {
441 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
442 return false;
443 bd->MousePendingLeaveFrame = ImGui::GetFrameCount() + 1;
444 return true;
445 }
446 case SDL_EVENT_WINDOW_FOCUS_GAINED:
447 case SDL_EVENT_WINDOW_FOCUS_LOST:
448 {
449 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == nullptr)
450 return false;
451 io.AddFocusEvent(event->type == SDL_EVENT_WINDOW_FOCUS_GAINED);
452 return true;
453 }
454 case SDL_EVENT_WINDOW_CLOSE_REQUESTED:
455 case SDL_EVENT_WINDOW_MOVED:
456 case SDL_EVENT_WINDOW_RESIZED:
457 {
458 ImGuiViewport* viewport = ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID);
459 if (viewport == NULL)
460 return false;
461 if (event->type == SDL_EVENT_WINDOW_CLOSE_REQUESTED)
462 viewport->PlatformRequestClose = true;
463 if (event->type == SDL_EVENT_WINDOW_MOVED)
464 viewport->PlatformRequestMove = true;
465 if (event->type == SDL_EVENT_WINDOW_RESIZED)
466 viewport->PlatformRequestResize = true;
467 return true;
468 }
469 case SDL_EVENT_GAMEPAD_ADDED:
470 case SDL_EVENT_GAMEPAD_REMOVED:
471 {
472 bd->WantUpdateGamepadsList = true;
473 return true;
474 }
475 }
476 return false;
477}

647// This code is incredibly messy because some of the functions we need for full viewport support are not available in SDL < 2.0.4.
648static void ImGui_ImplSDL3_UpdateMouseData()
649{
650 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
651 ImGuiIO& io = ImGui::GetIO();
652
653 // We forward mouse input when hovered or captured (via SDL_EVENT_MOUSE_MOTION) or when focused (below)
654#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
655 // - SDL_CaptureMouse() let the OS know e.g. that our drags can extend outside of parent boundaries (we want updated position) and shouldn't trigger other operations outside.
656 // - Debuggers under Linux tends to leave captured mouse on break, which may be very inconvenient, so to migitate the issue we wait until mouse has moved to begin capture.
657 bool want_capture = false;
658 for (int button_n = 0; button_n < ImGuiMouseButton_COUNT && !want_capture; button_n++)
659 if (ImGui::IsMouseDragging(button_n, 1.0f))
660 want_capture = true;
661 SDL_CaptureMouse(want_capture);
662
663 SDL_Window* focused_window = SDL_GetKeyboardFocus();
664 const bool is_app_focused = (focused_window && (bd->Window == focused_window || ImGui_ImplSDL3_GetViewportForWindowID(SDL_GetWindowID(focused_window)) != NULL));
665#else
666 SDL_Window* focused_window = bd->Window;
667 const bool is_app_focused = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) != 0; // SDL 2.0.3 and non-windowed systems: single-viewport only
668#endif
669 if (is_app_focused)
670 {
671 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when io.ConfigNavMoveSetMousePos is enabled by user)
672 if (io.WantSetMousePos)
673 {
674#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
675 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
676 SDL_WarpMouseGlobal(io.MousePos.x, io.MousePos.y);
677 else
678#endif
679 SDL_WarpMouseInWindow(bd->Window, io.MousePos.x, io.MousePos.y);
680 }
681
682 // (Optional) Fallback to provide mouse position when focused (SDL_EVENT_MOUSE_MOTION already provides this when hovered or captured)
683 const bool is_relative_mouse_mode = SDL_GetWindowRelativeMouseMode(bd->Window);
684 if (bd->MouseCanUseGlobalState && bd->MouseButtonsDown == 0 && !is_relative_mouse_mode)
685 {
686 // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
687 // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
688 float mouse_x, mouse_y;
689 int window_x, window_y;
690 SDL_GetGlobalMouseState(&mouse_x, &mouse_y);
691 if (!(io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable))
692 {
693 SDL_GetWindowPosition(focused_window, &window_x, &window_y);
694 mouse_x -= window_x;
695 mouse_y -= window_y;
696 }
697 io.AddMousePosEvent((float)mouse_x, (float)mouse_y);
698 }
699 }
700
701 // (Optional) When using multiple viewports: call io.AddMouseViewportEvent() with the viewport the OS mouse cursor is hovering.
702 // If ImGuiBackendFlags_HasMouseHoveredViewport is not set by the backend, Dear imGui will ignore this field and infer the information using its flawed heuristic.
703 // - [!] SDL backend does NOT correctly ignore viewports with the _NoInputs flag.
704 // Some backend are not able to handle that correctly. If a backend report an hovered viewport that has the _NoInputs flag (e.g. when dragging a window
705 // for docking, the viewport has the _NoInputs flag in order to allow us to find the viewport under), then Dear ImGui is forced to ignore the value reported
706 // by the backend, and use its flawed heuristic to guess the viewport behind.
707 // - [X] SDL backend correctly reports this regardless of another viewport behind focused and dragged from (we need this to find a useful drag and drop target).
708 if (io.BackendFlags & ImGuiBackendFlags_HasMouseHoveredViewport)
709 {
710 ImGuiID mouse_viewport_id = 0;
711 if (ImGuiViewport* mouse_viewport = ImGui_ImplSDL3_GetViewportForWindowID(bd->MouseWindowID))
712 mouse_viewport_id = mouse_viewport->ID;
713 io.AddMouseViewportEvent(mouse_viewport_id);
714 }
715}

846static void ImGui_ImplSDL3_UpdateMonitors()
847{
848 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
849 platform_io.Monitors.resize(0);
850
851 int display_count;
852 SDL_DisplayID* displays = SDL_GetDisplays(&display_count);
853 for (int n = 0; n < display_count; n++)
854 {
855 // Warning: the validity of monitor DPI information on Windows depends on the application DPI awareness settings, which generally needs to be set in the manifest or at runtime.
856 SDL_DisplayID display_id = displays[n];
857 ImGuiPlatformMonitor monitor;
858 SDL_Rect r;
859 SDL_GetDisplayBounds(display_id, &r);
860 monitor.MainPos = monitor.WorkPos = ImVec2((float)r.x, (float)r.y);
861 monitor.MainSize = monitor.WorkSize = ImVec2((float)r.w, (float)r.h);
862 if (SDL_GetDisplayUsableBounds(display_id, &r) && r.w > 0 && r.h > 0)
863 {
864 monitor.WorkPos = ImVec2((float)r.x, (float)r.y);
865 monitor.WorkSize = ImVec2((float)r.w, (float)r.h);
866 }
867 // FIXME-VIEWPORT: On MacOS SDL reports actual monitor DPI scale, ignoring OS configuration. We may want to set
868 // DpiScale to cocoa_window.backingScaleFactor here.
869 monitor.DpiScale = SDL_GetDisplayContentScale(display_id);
870 monitor.PlatformHandle = (void*)(intptr_t)n;
871 if (monitor.DpiScale <= 0.0f)
872 continue; // Some accessibility applications are declaring virtual monitors with a DPI of 0, see #7902.
873 platform_io.Monitors.push_back(monitor);
874 }
875 SDL_free(displays);
876}

884 // Setup display size (every frame to accommodate for window resizing)
885 int w, h;
886 int display_w, display_h;
887 SDL_GetWindowSize(bd->Window, &w, &h);
888 if (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_MINIMIZED)
889 w = h = 0;
890 SDL_GetWindowSizeInPixels(bd->Window, &display_w, &display_h);
891 io.DisplaySize = ImVec2((float)w, (float)h);
892 if (w > 0 && h > 0)
893 io.DisplayFramebufferScale = ImVec2((float)display_w / w, (float)display_h / h);
894
895 // Update monitors
896 ImGui_ImplSDL3_UpdateMonitors();
897
898 // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
899 // (Accept SDL_GetPerformanceCounter() not returning a monotonically increasing value. Happens in VMs and Emscripten, see #6189, #6114, #3644)
900 static Uint64 frequency = SDL_GetPerformanceFrequency();
901 Uint64 current_time = SDL_GetPerformanceCounter();
902 if (current_time <= bd->Time)
903 current_time = bd->Time + 1;
904 io.DeltaTime = bd->Time > 0 ? (float)((double)(current_time - bd->Time) / frequency) : (float)(1.0f / 60.0f);
905 bd->Time = current_time;
906
907 if (bd->MousePendingLeaveFrame && bd->MousePendingLeaveFrame >= ImGui::GetFrameCount() && bd->MouseButtonsDown == 0)
908 {
909 bd->MouseWindowID = 0;
910 bd->MousePendingLeaveFrame = 0;
911 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
912 }
913
914 // Our io.AddMouseViewportEvent() calls will only be valid when not capturing.
915 // Technically speaking testing for 'bd->MouseButtonsDown == 0' would be more rigorous, but testing for payload reduces noise and potential side-effects.
916 if (bd->MouseCanReportHoveredViewport && ImGui::GetDragDropPayload() == nullptr)
917 io.BackendFlags |= ImGuiBackendFlags_HasMouseHoveredViewport;
918 else
919 io.BackendFlags &= ~ImGuiBackendFlags_HasMouseHoveredViewport;
920
921 ImGui_ImplSDL3_UpdateMouseData();
922 ImGui_ImplSDL3_UpdateMouseCursor();
923
924 // Update game controllers (if enabled and available)
925 ImGui_ImplSDL3_UpdateGamepads();
926}

959{
960 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();

976 SDL_GL_MakeCurrent(main_viewport_data->Window, main_viewport_data->GLContext);
977 }

980 sdl_flags |= SDL_WINDOW_HIDDEN;
981 sdl_flags |= use_opengl ? SDL_WINDOW_OPENGL : (bd->UseVulkan ? SDL_WINDOW_VULKAN : 0);
982 sdl_flags |= SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_HIGH_PIXEL_DENSITY;
983 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoDecoration) ? SDL_WINDOW_BORDERLESS : 0;
984 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoDecoration) ? 0 : SDL_WINDOW_RESIZABLE;
985 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoTaskBarIcon) ? SDL_WINDOW_UTILITY : 0;
986 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_TopMost) ? SDL_WINDOW_ALWAYS_ON_TOP : 0;
987 vd->Window = SDL_CreateWindow("No Title Yet", (int)viewport->Size.x, (int)viewport->Size.y, sdl_flags);
988 SDL_SetWindowParent(vd->Window, vd->ParentWindow);
989 SDL_SetWindowPosition(vd->Window, (int)viewport->Pos.x, (int)viewport->Pos.y);
990 vd->WindowOwned = true;
991 if (use_opengl)
992 {
993 vd->GLContext = SDL_GL_CreateContext(vd->Window);
994 SDL_GL_SetSwapInterval(0);
995 }
996 if (use_opengl && backup_context)
997 SDL_GL_MakeCurrent(vd->Window, backup_context);

999 ImGui_ImplSDL3_SetupPlatformHandles(viewport, vd->Window);
1000}
